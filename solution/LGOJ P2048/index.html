LGOJ #P2048. 「NOI2010」超级钢琴

内存限制：512 MiB 时间限制：2000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

小 Z 是一个小有名气的钢琴家，最近 C 博士送给了小 Z 一架超级钢琴，小 Z 希望能够用这架钢琴创作出世界上最美妙的音乐。

这架超级钢琴可以弹奏出 $n$ 个音符，编号为 $1$ 至 $n$。第 $i$ 个音符的美妙度为 $A_i$，其中 $A_i$ 可正可负。

一个 “超级和弦” 由若干个编号连续的音符组成，包含的音符个数不少于 $L$ 且不多于 $R$。我们定义超级和弦的美妙度为其包含的所有音符的美妙度之和。两个超级和弦被认为是相同的，当且仅当这两个超级和弦所包含的音符集合是相同的。

小 Z 决定创作一首由 $k$ 个超级和弦组成的乐曲，为了使得乐曲更加动听，小 Z 要求该乐曲由 $k$ 个不同的超级和弦组成。我们定义一首乐曲的美妙度为其所包含的所有超级和弦的美妙度之和。小 Z 想知道他能够创作出来的乐曲美妙度最大值是多少。

输入格式

输入第一行包含四个正整数 $n, k, L, R$。其中 $n$ 为音符的个数，$k$ 为乐曲所包含的超级和弦个数，$L$ 和 $R$ 分别是超级和弦所包含音符个数的下限和上限。

接下来 $n$ 行，每行包含一个整数 $A_i$，表示按编号从小到大每个音符的美妙度。

输出格式

输出只有一个整数，表示乐曲美妙度的最大值。

样例

样例输入

4 3 2 3
3
2
-6
8

样例输出

11

数据范围与提示

共有 $5$ 种不同的超级和弦：

  1. 音符 $1 \sim 2$，美妙度为 $3 + 2 = 5$
  2. 音符 $2 \sim 3$，美妙度为 $2 + (-6) = -4$
  3. 音符 $3 \sim 4$，美妙度为 $(-6) + 8 = 2$
  4. 音符 $1 \sim 3$，美妙度为 $3 + 2 + (-6) = -1$
  5. 音符 $2 \sim 4$，美妙度为 $2 + (-6) + 8 = 4$

最优方案为：乐曲由和弦 $1$，和弦 $3$，和弦 $5$ 组成，美妙度为 $5 + 2 + 4 = 11$。

$3$ $\leq 100,000$ $=1$

所有数据满足：$-1000 \leq A_i \leq 1000$，$1 \leq L \leq R \leq n$ 且保证一定存在满足要求的乐曲。

题解

明显，可以差分一下，这样每个区间的和就变成了 $S_r - S_{l - 1}$。

可持久化线段树。

我们用第 $i$ 棵线段树维护 $\max\{S_r\}$，其中 $[i, r]$ 是合法区间。当取出该区间时，把第 $i$ 棵线段树上的 $S_r$ 置为 $-\infty$ 即可。

时间复杂度：$O((n+k) \log n)$。

代码

#include <cstdio>
#include <queue>

#define mid ((l + r) >> 1)
#define lp lson[p]
#define rp rson[p]
#define lc lson[p], l, mid
#define rc rson[p], mid + 1, r

typedef long long int64;

#ifdef ONLINE_JUDGE
const int MAXN = 500000 + 10;
const int MAXSIZE = 20000000;
#else
const int MAXN = 1000;
const int MAXSIZE = 1000;
#endif

const int INF = 0x7fffffff;

int n, m, lb, rb;
int sum[MAXN];
int rt[MAXN];
int maxp[MAXSIZE], lson[MAXSIZE], rson[MAXSIZE], nodecnt;

struct Node {
    int p, v;

    Node() {}
    Node(int _p, int _v) {
        p = _p;
        v = _v;
    }
    Node operator=(const Node b) {
        p = b.p;
        v = b.v;
        return *this;
    }
    bool operator<(const Node b) const { return v < b.v; }
};
std::priority_queue<Node> pq;

inline void up(int p) { maxp[p] = sum[maxp[lp]] > sum[maxp[rp]] ? maxp[lp] : maxp[rp]; }
void ins(int &p, int l, int r, int a, int b) {
    nodecnt++;
    lson[nodecnt] = lson[p];
    rson[nodecnt] = rson[p];
    p = nodecnt;
    if (l == r) {
        maxp[p] = b;
        return;
    }
    else {
        if (a <= mid)
            ins(lc, a, b);
        else
            ins(rc, a, b);
        up(p);
    }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("P2048.in", "r", stdin);
    freopen("P2048.out", "w", stdout);
#endif

    int i;
    int64 ans = 0;
    Node x;

    scanf("%d %d %d %d", &n, &m, &lb, &rb);
    for (i = 1; i <= n; i++) {
        scanf("%d", &sum[i]);
        sum[i] += sum[i - 1];
    }
    sum[0] = -INF;
    for (i = n - lb + 1; i; i--) {
        rt[i] = rt[i + 1];
        if (i == 27904) {
            printf("\n");
        }
        ins(rt[i], 1, n, i + lb - 1, i + lb - 1);
        if (i + rb <= n)
            ins(rt[i], 1, n, i + rb, 0);
        pq.push(Node(i, sum[maxp[rt[i]]] - bool(i - 1) * sum[i - 1]));
    }

    for (i = 1; i <= m; i++) {
        x = pq.top();
        pq.pop();
        ans += x.v;
        x.v = maxp[rt[x.p]];
        ins(rt[x.p], 1, n, maxp[rt[x.p]], 0);
        if (maxp[rt[x.p]] == x.v)
            printf("%d %d Error\n", x.p, x.v);
        if (maxp[rt[x.p]])
            pq.push(Node(x.p, sum[maxp[rt[x.p]]] - bool(x.p - 1) * sum[x.p - 1]));
    }
    printf("%lld\n", ans);
    return 0;
}
