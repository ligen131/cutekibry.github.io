LGOJ #P2000. 拯救世界

内存限制：128 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

公元 $2000$ 年，根据预言家诺查丹玛斯的预言，世界就要毁灭了！！！

为了拯救世界，小 a 和 uim 决定召唤出 kkksc03 大神和 lzn 大神。根据古籍记载，召唤出任何一位大神，都需要使用金木水火土五种五行神石来摆一个特定的大阵。而在古籍中，记载是这样的：

kkksc03 大神召唤方法：

金神石的块数必须是 $6$ 的倍数。

木神石最多用 $9$ 块。

水神石最多用 $5$ 块。

火神石的块数必须是 $4$ 的倍数。

土神石最多用 $7$ 块。

lzn 大神召唤方法:

金神石的块数必须是 $2$ 的倍数。

木神石最多用 $1$ 块。

水神石的块数必须是 $8$ 的倍数。

火神石的块数必须是 $10$ 的倍数。

土神石最多用 $3$ 块。

现在是公元 $1999$ 年 $12$ 月 $31$ 日，小 a 和 uim 从 $00:00:00$ 开始找，一直找到 $23:00:00$，终于，还是没找到神石。不过，他们在回到家后在自家地窖里发现了一些奇怪的东西，一查古籍，哎呦妈呀，怎么不早点来呢？这里有一些混沌之石，可以通过敲击而衰变成五行神石。于是，他们拼命地敲，终于敲出了 $n$ 块神石，在 $23:59:59$ 完成了两座大阵。然而，kkksc03 大神和 lzn 大神确实出现了，但是由于能量不够，无法发挥神力。只有把所有用 $n$ 块神石可能摆出的大阵都摆出来，才能给他们充满能量。这下小 a 和 uim 傻了眼了，赶快联系上了你，让你帮忙算一下，一共有多少种大阵。

输入格式

输入一个数 $n$。

输出格式

输出用 $n$ 块混沌之石能摆出的大阵的种数。

样例

样例输入

2

样例输出

15

数据范围与提示

数据范围：

$length(n)=100000$

是等于！不是小于等于！

提示 由于现在已经是 $23:59:59$，所以你只有 0.5s 时间。（小 a 与 uim 需要 0.5s 排出所有阵法）

题解

题目大意就是求在这 $10$ 种物品中选择 $n$ 个物品的方案数。

直接上生成函数，可得：

$$ \begin{aligned} g(x) &= \frac 1 {1 - x^2} \frac 1 {1 - x^4} \frac 1 {1 - x^6} \frac 1 {1 - x^8} \frac 1 {1 - x^{10}} \frac {1 - x^2} {1 - x} \frac {1 - x^4} {1 - x} \frac {1 - x^6} {1 - x} \frac {1 - x^8} {1 - x} \frac {1 - x^{10}} {1 - x} \\ &= \frac {1} {(1 - x)^5} \\ &= \mathrm{C}^4_4 + \mathrm{C}^4_5 x+ \mathrm{C}^4_6 x^2 + \mathrm{C}^4_7 x^3 + \ldots\\ \end{aligned} $$

所以可得 $n$ 的方案数为 $C^4_{4+n} = \frac {(n + 1)(n + 2)(n + 3)(n + 4)}{24}$。

高精度 FFT/NTT 即可。

时间复杂度：$O(\lg n \log \lg n)$。

代码

// luogu-judger-enable-o2
#include <algorithm>
#include <cstdio>
#include <cstring>

typedef long long int64;

const int G = 3;
const int INVG = 332748118;
const int MOD = 998244353;
const int LEN = 1 << 19;
const int L = 19;
const int INVLEN = 998242449;

typedef int Expr[LEN];

int rev[LEN];

int invw[LEN * L], w[LEN * L];

inline int qmod(int x, int mod) {
    while (x >= mod)
        x -= mod;
    return x;
}
inline int qpow(int b, int p, int mod) {
    int res = 1;
    while (p) {
        if (p & 1)
            res = (int64)res * b % mod;
        b = (int64)b * b % mod;
        p >>= 1;
    }
    return res;
}

void init() {
    int i, j, k, p;
    int wn, ww;

    for (i = 1; i < LEN; i++)
        rev[i] = (rev[i >> 1] >> 1) | ((i & 1) << (L - 1));
    p = 0;
    for (k = 1; k < LEN; k <<= 1) {
        wn = qpow(G, (MOD - 1) / (k << 1), MOD);
        for (i = 0; i < LEN; i += k << 1) {
            ww = 1;
            for (j = 0; j < k; j++) {
                p++;
                w[p] = ww;
                ww = (int64)ww * wn % MOD;
            }
        }
    }
    p = 0;
    for (k = 1; k < LEN; k <<= 1) {
        wn = qpow(INVG, (MOD - 1) / (k << 1), MOD);
        for (i = 0; i < LEN; i += k << 1) {
            ww = 1;
            for (j = 0; j < k; j++) {
                p++;
                invw[p] = ww;
                ww = (int64)ww * wn % MOD;
            }
        }
    }
}
void ntt(Expr a, bool inv) {
    int i, j, k, p = 0, x, y;

    for (i = 1; i < LEN; i++)
        if (i < rev[i])
            std::swap(a[i], a[rev[i]]);
    for (k = 1; k < LEN; k <<= 1) {
        for (i = 0; i < LEN; i += k << 1) {
            for (j = 0; j < k; j++) {
                p++;
                x = a[i + j];
                y = (int64)a[i + j + k] * (inv ? invw[p] : w[p]) % MOD;
                a[i + j] = qmod(x + y, MOD);
                a[i + j + k] = qmod(x - y + MOD, MOD);
            }
        }
    }
    if (inv)
        for (i = 0; i < LEN; i++)
            a[i] = (int64)a[i] * INVLEN % MOD;
}

Expr a, b, c, d;

inline void assign(Expr a, Expr b) { memcpy(a, b, sizeof(Expr)); }
inline void read(Expr a) {
    int i, j, len = 0;
    char ch;

    memset(a, 0, sizeof(Expr));
    do
        ch = getchar();
    while (ch < '0' or ch > '9');
    while (ch >= '0' and ch <= '9') {
        a[len] = ch & 15;
        len++;
        ch = getchar();
    }
    for (i = 0, j = len - 1; i < j; i++, j--)
        std::swap(a[i], a[j]);
}
inline void writeln(Expr a) {
    int i;

    i = LEN - 1;
    while (!a[i])
        i--;
    while (~i) {
        putchar(a[i] | 48);
        i--;
    }
    putchar('\n');
}
inline void inc(Expr a) {
    int i;

    for (i = 0; a[i] == 9; i++)
        a[i] = 0;
    a[i]++;
}
inline void mul(Expr a, Expr b) {
    int i;

    for (i = 0; i < LEN; i++)
        a[i] = (int64)a[i] * b[i] % MOD;
}
inline void div(Expr a, int b) {
    int i, rest = 0;

    for (i = LEN - 1; ~i; i--) {
        rest = rest * 10 + a[i];
        a[i] = rest / b;
        rest %= b;
    }
}
inline void adjust(Expr a) {
    int i;

    for (i = 0; i < LEN - 1; i++) {
        if (a[i] >= 10) {
            a[i + 1] += a[i] / 10;
            a[i] %= 10;
        }
    }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("P2000.in", "r", stdin);
    freopen("P2000.out", "w", stdout);
#endif

    init();

    read(a);
    inc(a);
    assign(b, a);
    inc(b);
    assign(c, b);
    inc(c);
    assign(d, c);
    inc(d);

    ntt(a, false);
    ntt(b, false);
    ntt(c, false);
    ntt(d, false);

    mul(a, b);
    ntt(a, true);
    adjust(a);
    ntt(a, false);

    mul(a, c);
    ntt(a, true);
    adjust(a);
    ntt(a, false);

    mul(a, d);
    ntt(a, true);
    adjust(a);
    div(a, 24);
    writeln(a);

    return 0;
}
