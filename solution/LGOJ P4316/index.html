LGOJ #P4316. 绿豆蛙的归宿

内存限制：128 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

随着新版百度空间的上线，Blog 宠物绿豆蛙完成了它的使命，去寻找它新的归宿。

给出一个有向无环图，起点为 $1$，终点为 $N$，每条边都有一个长度，并且从起点出发能够到达所有的点，所有的点也都能够到达终点。绿豆蛙从起点出发，走向终点。

到达每一个顶点时，如果有 $K$ 条离开该点的道路，绿豆蛙可以选择任意一条道路离开该点，并且走向每条路的概率为 $\frac 1K$。

现在绿豆蛙想知道，从起点走到终点的所经过的路径总长度期望是多少？

输入格式

第一行，两个整数 $N, M$，代表图中有 $N$ 个点，$M$ 条边。

接下来 $M$ 行，每行 $3$ 个整数 $a, b, c$，代表从 $a$ 到 $b$ 有一条长度为 $c$ 的有向边。

输出格式

从起点到终点路径总长度的期望值，四舍五入保留两位小数。

样例

样例输入

4 4
1 2 1
1 3 2
2 3 3
3 4 4

样例输出

7.00

数据范围与提示

对于 $20\%$ 的数据，$N \leq 100$；

对于 $40\%$ 的数据，$N \leq 1000$；

对于 $60\%$ 的数据，$N \leq 10000$；

对于 $100\%$ 的数据，$N \leq 100000，M \leq 2N$。

题解

主页面：「专题」随机游走

直接做即可。

时间复杂度：$O(n + m)$。

代码

方法一：依照题意设状态

#include <cstdio>

const int MAXN = 200000 + 10;

int n, m;
int pre[MAXN], to[MAXN], wdis[MAXN], head[MAXN], wcnt;
int odeg[MAXN];
double f[MAXN];

inline void addedge(int u, int v, int w) {
    wcnt++;
    pre[wcnt] = head[u];
    head[u] = wcnt;
    to[wcnt] = v;
    wdis[wcnt] = w;
}

#define v to[i]
#define w wdis[i]
void dfs(int u) {
    static bool vis[MAXN];
    int i;

    if (vis[u])
        return;
    vis[u] = true;
    for (i = head[u]; i; i = pre[i]) {
        dfs(v);
        f[u] += f[v] + w;
    }
    if (odeg[u])
        f[u] /= odeg[u];
}
#undef w
#undef v

int main() {
#ifndef ONLINE_JUDGE
    freopen("P4316.in", "r", stdin);
    freopen("P4316-usual.out", "w", stdout);
#endif

    int u, v, w;
    int i;

    scanf("%d %d", &n, &m);
    for (i = 1; i <= m; i++) {
        scanf("%d %d %d", &u, &v, &w);
        addedge(u, v, w);
        odeg[u]++;
    }
    dfs(1);
    printf("%.2lf\n", f[1]);
    return 0;
}

方法二：计算边经过的期望次数

#include <cstdio>

const int MAXN = 200000 + 10;

int n, m;
int pre[MAXN], to[MAXN], wdis[MAXN], head[MAXN], wcnt;
int odeg[MAXN];
double f[MAXN];

inline void addedge(int u, int v, int w) {
    wcnt++;
    pre[wcnt] = head[u];
    head[u] = wcnt;
    to[wcnt] = v;
    wdis[wcnt] = w;
}

#define v to[i]
void dfs(int u) {
    static bool vis[MAXN];
    int i;

    if (vis[u])
        return;
    vis[u] = true;

    if (u == 1) {
        f[u] = 1;
        return;
    }
    for (i = head[u]; i; i = pre[i]) {
        dfs(v);
        f[u] += f[v] / odeg[v];
    }
    printf("%d %.2lf\n", u, f[u]);
}
#undef v

int main() {
#ifndef ONLINE_JUDGE
    freopen("P4316.in", "r", stdin);
    freopen("P4316-edge.out", "w", stdout);
#endif

    int u, v, w;
    int i;
    double ans = 0;

    scanf("%d %d", &n, &m);
    for (i = 1; i <= m; i++) {
        scanf("%d %d %d", &u, &v, &w);
        addedge(v, u, w);
        odeg[u]++;
    }
    for (i = 1; i <= n; i++)
        dfs(i);

    for (i = 1; i <= m; i++)
        ans += wdis[i] * f[to[i]] / odeg[to[i]];
    printf("%.2lf\n", ans);
    return 0;
}
