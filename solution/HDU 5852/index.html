HDU #5852. Intersection is not allowed!

内存限制：64 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

棋盘上有 $K$ 个棋子。

棋盘的大小是 $N \times N$。

一开始，棋子被放在棋盘最上面的格子里。

棋子在棋盘上的任何一个格子 $(r, c)$ 可以向右移动一格到 $(r, c + 1)$ 或向下移动一格到 $(r + 1, c)$。

你的任务是计算将所有棋子移动到最下面给定位置的方案数。

此外，路径间必须两两不相交。

输入格式

第一行，一个整数 $T$，表示数据组数。

每组数据的第一行，两个整数 $N, K$，分别表示棋盘大小和棋子个数。

第二行，$K$ 个整数 $1 \leq a_1 < a_2 < \ldots < a_K \leq N$，表示初始位置。也就是说，棋子一开始被放在 $(1, a_1), (1, a_2), \ldots, (1, a_K)$。

下一行，$K$ 个整数 $1 \leq b_1 < b_2 < \ldots < b_K \leq N$，表示最终位置。也就是说，棋子应当被移动到 $(N, b_1), (N, b_2), \ldots, (N, b_K)$。

输出格式

输出连续 $T$ 行，分别表示方案数对 $1000000007$ 取模后的值。

样例

样例输入

1
5 2
1 2
3 4

样例输出

50

题解

LGV 定理的模板题。

从 $(0, 0)$ 走到 $(n, m)$ 的方案数为 $\mathrm{C}^n_{n + m}$。走到 $(n, m)$ 需要走 $n + m$ 步，在这之中需要向右走 $n$ 步，所以是 $\mathrm{C}^n_{n + m}$。

数据保证了 $a_i \leq b_i$（尽管题目中并没有说），所以即使不判断方案是否为 $0$ 也可以通过本题。

当然，按照本题要求是应该特判的。

时间复杂度：$O(N + TK^3)$。

代码

#include <algorithm>
#include <cstdio>

typedef long long int64;

const int MOD = 1000000007;

#ifdef ONLINE_JUDGE
const int MAXN = 200000 + 10;
const int MAXK = 100 + 10;
#else
const int MAXN = 1000 + 10;
const int MAXK = 30 + 10;
#endif

int fac[MAXN], invfac[MAXN];
int a[MAXK][MAXK];
int s[MAXK], t[MAXK];
int n, m;

inline int qpow(int64 b, int p, int mod) {
    int res = 1;
    while (p) {
        if (p & 1)
            res = res * b % mod;
        b = b * b % mod;
        p >>= 1;
    }
    return res;
}
inline int C(int n, int m) {
    if (n > m)
        return 0;
    else
        return 1ll * fac[m] * invfac[n] % MOD * invfac[m - n] % MOD;
}

void init() {
    int i;

    fac[0] = invfac[0] = 1;
    for (i = 1; i < MAXN; i++)
        fac[i] = 1ll * fac[i - 1] * i % MOD;

    invfac[MAXN - 1] = qpow(fac[MAXN - 1], MOD - 2, MOD);
    for (i = MAXN - 2; i; i--)
        invfac[i] = invfac[i + 1] * (i + 1ll) % MOD;
}
int solve() {
    int i, j, k;
    int64 p, inva;
    bool neg = false;
    int64 ans = 1;

    for (i = 1; i < m; i++) {
        k = i;
        while (k <= m and !a[k][i])
            k++;
        if (k > m)
            return 0;
        if (i ^ k) {
            for (j = i; j <= m; j++)
                std::swap(a[i][j], a[k][j]);
            neg = !neg;
        }

        inva = qpow(MOD - a[i][i], MOD - 2, MOD);
        for (j = i + 1; j <= m; j++) {
            p = a[j][i] * inva % MOD;
            for (k = i; k <= m; k++)
                a[j][k] = (a[j][k] + p * a[i][k]) % MOD;
        }
    }
    for (i = 1; i <= m; i++)
        ans = ans * a[i][i] % MOD;
    if (neg and ans)
        ans = MOD - ans;
    return ans;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("5852.in", "r", stdin);
    freopen("5852.out", "w", stdout);
#endif

    int i, j;
    int T;
    bool flag;

    init();

    scanf("%d", &T);
    while (T) {
        scanf("%d %d", &n, &m);
        for (i = 1; i <= m; i++)
            scanf("%d", &s[i]);
        for (i = 1; i <= m; i++)
            scanf("%d", &t[i]);

        flag = false;
        for (i = 1; i <= m; i++)
            for (j = 1; j <= m; j++)
                if (i ^ j and s[i] < s[j] and t[i] > t[j])
                    flag = true;
        if (flag)
            printf("0\n");
        else {
            for (i = 1; i <= m; i++)
                for (j = 1; j <= m; j++)
                    a[i][j] = C(n - 1, n - 1 + t[j] - s[i]);
            printf("%d\n", solve());
        }
        T--;
    }
}
