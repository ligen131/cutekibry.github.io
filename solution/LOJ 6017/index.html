LOJ #6017. Shlw loves matrix I

内存限制：256 MiB 时间限制：2000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

给定数列 $\{h_n\}$ 前 $k$ 项，其后每一项满足

$$h_n = a_1 \times h_{n-1} + a_2 \times h_{n-2} + ... + a_k \times h_{n-k}$$

其中 $a_1,a_2\cdots a_k$ 为给定数列。请计算 $h(n)$，并将结果对 $1000000007$ 取模输出。

输入格式

第一行输入两个正整数 $n,k$

第二行输入 $k$ 个正整数表示 $a_1,\dots, a_k$

第三行输入 $k$ 个正整数表示 $h_0,\cdots ,h_{k-1}$

输出格式

一行输出一个数，表示 $h_n$ 对 $1000000007$ 取模的结果。

样例

样例输入

6 4
3 -1 0 4
-2 3 1 5

样例输出

73

数据范围与提示

$k\leq 2000,n\leq 10^9$

题解

主页面：「专题」线性递推

模板题。

数据范围较小，所以可以直接用快速幂计算。

时间复杂度：$O(k^2 \log n)$。

代码

#include <cstdio>
#include <cstring>

const int MOD = 1000000007;
const int MAXN = 4000 + 10;

typedef int Expr[MAXN];

int n, m;
Expr hh, aa, b, res, mod;

inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }

inline void mul(Expr a, Expr b, Expr c) {
    static Expr res;
    int i, j, p;

    memset(res, 0, sizeof(Expr));
    for (i = 0; i < m; i++)
        for (j = 0; j < m; j++)
            res[i + j] = (res[i + j] + 1ll * b[i] * c[j]) % MOD;

    for (i = (m - 1) << 1; i >= m; i--) {
        if (res[i]) {
            for (j = 0; j <= m; j++)
                res[i - j - 1] = (res[i - j - 1] + 1ll * res[i] * aa[j]) % MOD;
            res[i] = 0;
        }
    }
    memcpy(a, res, sizeof(Expr));
}
int main() {
#ifndef ONLINE_JUDGE
    freopen("6017.in", "r", stdin);
    freopen("6017.out", "w", stdout);
#endif

    int i, ans;

    scanf("%d %d", &n, &m);
    for (i = 0; i < m; i++) {
        scanf("%d", &aa[i]);
        if (aa[i] < 0)
            aa[i] += MOD;
    }
    for (i = 0; i < m; i++) {
        scanf("%d", &hh[i]);
        if (hh[i] < 0)
            hh[i] += MOD;
    }

    res[0] = 1;
    b[1] = 1;
    while (n) {
        if (n & 1)
            mul(res, res, b);
        mul(b, b, b);
        n >>= 1;
    }

    ans = 0;
    for (i = 0; i < m; i++)
        ans = (ans + 1ll * res[i] * hh[i]) % MOD;
    printf("%d\n", ans);
    return 0;
}
