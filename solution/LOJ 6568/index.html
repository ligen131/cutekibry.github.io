LOJ #6568. 「Project Euler」特殊勾股数

题目描述

一组勾股数由三个自然数组成，$a < b < c$，且

$$a^2 + b^2 = c^2$$

例如，$3^2 + 4^2 = 9 + 16 = 25 = 5^2$。

给出 $N$，请输出所有满足 $a + b + c = N$ 的勾股数。

输入格式

只有一行，一个整数 $N$。

输出格式

多行，每行三个整数，$a, b, c$，表示一组勾股数。

样例

样例输入 1

12

样例输出 1

3 4 5

样例输入 2

120

样例输出 2

30 40 50
20 48 52
24 45 51

数据范围与提示

对于 $20\%$ 的数据，$N \leq 5000$。

对于 $40\%$ 的数据，$N \leq 1000000$。

对于 $100\%$ 的数据，$N \leq 10^{12}$。

题解

$20\%$

枚举 $a, b$，可得 $c = N - a - b$。

时间复杂度：$O(N^2)$。

$40\%$

$$ \begin{aligned} a^2 + b^2 &= (N - (a + b))^2 \\ a^2 + b^2 &= N^2 - 2n(a + b) + (a + b)^2 \\ a^2 + b^2 &= N^2 - 2an - 2bn + a^2 + 2ab + b^2 \\ -2ab &= N^2 - 2an - 2bn \\ 2ab &= 2n(a + b) - N^2 \\ \end{aligned} $$

枚举 $a$，可得

$$ \begin{aligned} (2a - 2n)b &= 2na - N^2 \\ b &= \frac{2na - N^2}{2a - 2n} \\ \end{aligned} $$

时间复杂度：$O(N)$。

$100\%$

定义：满足 $a^2 + b^2 = c^2$ 且 $a, b, c$ 均为正整数的三元组 $(a, b, c)$ 为勾股数。

若 $\gcd(a, b, c) = 1$，那么我们称此勾股数为质数勾股数。

易证，任意质数勾股数都可表示为 $(m^2 - n^2, 2mn, m^2 + n^2)$。枚举任意 $m > n > 0$，可以得到多组勾股数。

但这样生成出来的勾股数不一定是质数勾股数——当且仅当 $n, m$ 中有且仅有 $1$ 个数是偶数，且 $\gcd(n, m)=1$，这样的勾股数才是质数勾股数。

对于任意勾股数，我们可以设 $\gcd(a, b, c)=d$，并将其表示为 $((m^2 - n^2)d, 2mnd, (m^2 + n^2)d)$。

用这种方法，我们可以发现 $a + b + c = 2m(m + n)d = N$。

因此，我们可以考虑枚举 $m$ 和 $(m + n)$。

时间复杂度：$\approx O(\sqrt N)$。

代码

#include <algorithm>
#include <cstdio>

typedef long long int64;

int gcd(int x, int y) { return y ? gcd(y, x % y) : x; }

void solve(int64 *a, int64 n) {
    int64 i;

    a[0] = 0;
    for (i = 1; i * i <= n; i++) {
        if (!(n % i)) {
            a[0]++;
            a[a[0]] = i;
            if (i * i < n) {
                a[0]++;
                a[a[0]] = n / i;
            }
        }
    }
}

int main() {
// #ifndef ONLINE_JUDGE
//     freopen("sample.in", "r", stdin);
//     freopen("6568.out", "w", stdout);
// #endif

    static int64 dn[10005];
    static int64 dnn[10005];
    int64 N, NN, n, m, d, a, b, c;
    int i, j;

    scanf("%lld", &N);
    if (N & 1) {
        printf("0\n");
        return 0;
    }

    N >>= 1;
    solve(dn, N);
    for (i = 1; i <= dn[0]; i++) {
        m = dn[i];
        NN = N / m;
        solve(dnn, NN);
        for (j = dnn[0]; j; j--) {
            n = dnn[j] - m;
            if (m > n and n > 0 and gcd(m, n) == 1 and (n + m) & 1) {
                // printf("%lld %lld\n", m, n);
                d = N / m / (m + n);
                a = (m * m - n * n) * d;
                b = 2 * m * n * d;
                if (a > b)
                    std::swap(a, b);
                c = (m * m + n * n) * d;
                printf("%lld %lld %lld\n", a, b, c);
            }
        }
    }
    return 0;
}
