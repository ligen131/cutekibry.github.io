CF #799F. Beautiful fountains rows

内存限制：256 MiB 时间限制：3000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知

题目描述

有一个 $n \times m$ 的 $01$ 矩阵 $A$。对于第 $i$ 行，只有下标在 $[l_i, r_i]$ 中的 $A_{ij}$ 为 $1$。

定义一个区间 $[l, r]$ 是合法的，当且仅当对于任意 $1 \leq i \leq n$，$\sum_{j = l}^r A_{ij}$ 为 $0$ 或为奇数，且至少有一行 $A_{ij}$ 的和不为 $0$。该区间的长度为 $r - l + 1$。

求所有合法区间的长度之和。

输入格式

第一行，两个整数 $n, m \: (1 \leq n, m \leq 2 \times 10^5)$。

接下来 $n$ 行，每行两个整数 $l_i, r_i \: (1 \leq l_i \leq r_i \leq m)$。

输出格式

只有一行，一个整数，表示答案。

样例

样例输入 1

1 5
2 4

样例输出 1

23

样例解释 1

合法的区间有：$(a, b): (1, 2), (1, 4), (1, 5), (2, 2), (2, 4), (2, 5), (3, 3), (4, 4), (4, 5)$。

样例输入 2

3 6
2 4
3 6
4 4

样例输出 2

19

样例解释 2

合法的区间有：$(a, b): (1, 2), (1, 5), (2, 2), (2, 5), (3, 3), (4, 4), (4, 6), (5, 5), (6, 6)$。

题解

容易注意到，如果将每一行看成一种数，那么合法的区间中，每种出现的数都一定出现了奇数次。

对于一类区间限制为「同一种数出现偶数次」的问题，可以考虑对每种数随机赋予一个权值 $a_i$。那么，对于任意合法区间，区间内所有数的异或值必定等于 $0$；反之不一定成立。不过在 $a_i$ 相当大且近似随机的情况下，我们可以认为这个算法的正确率是可观的。

本题要求出现奇数次，那只需要用「区间所有数的异或值」异或上「区间出现过的数的异或值」，当它等于 $0$ 时，我们可以认为它有极大概率是合法的；否则它一定不合法。

设 $S_i$ 为序列的前缀异或值，$f(i, j)$ 为区间 $[i, j]$ 出现过数的异或值，则有

$$S_r \oplus S_{l - 1} \oplus f(l, r) = 0$$

考虑如何求 $f(i, j)$。

直接考虑在查询区间内的四种情况太复杂了，不如考虑不在查询区间内的两种情况。

设 $F_i$ 表示右端点小于等于 $i$ 的线段异或值，$G_i$ 表示左端点大于等于 $i$ 的线段异或值，$allsum$ 表示所有线段的线段异或值，则 $f(i, j) = allsum \oplus F_{i - 1} \oplus G_{j + 1}$。

因此有

$$ \begin{aligned} S_r \oplus S_{l - 1} \oplus f(l, r) &= 0 \\ S_r \oplus S_{l - 1} \oplus F_{l - 1} \oplus G_{r + 1} \oplus allsum &= 0 \\ S_{l - 1} \oplus F_{l - 1} &= S_r \oplus G_{r + 1} \oplus allsum \\ \end{aligned} $$

枚举右端点，用 Hash 表维护 $S_{l - 1} \oplus F_{i - 1}$，查询一下即可。

注意空的区间也被计数了，因此最后还要减去空区间的贡献。

时间复杂度：$O(n + m)$。

代码

#include <algorithm>
#include <cstdio>
#include <cstring>
#include <map>

typedef unsigned long long uint64;

#ifdef ONLINE_JUDGE
const int MAXN = 200000 + 10;
#else
const int MAXN = 1000 + 10;
#endif

const int HMOD = 10000019;

int n, m;
int l[MAXN], r[MAXN];
uint64 a[MAXN], s[MAXN], allsum;
uint64 f[MAXN], g[MAXN];

uint64 key[MAXN], m0[MAXN], m1[MAXN];
int pre[MAXN], head[HMOD], nodecnt;

inline int gethash(uint64 k) { return k % HMOD; }
inline int find(uint64 k, bool create) {
    int h = gethash(k), p = head[h];
    while (p and key[p] ^ k)
        p = pre[p];
    if (!p and create) {
        nodecnt++;
        pre[nodecnt] = head[h];
        key[nodecnt] = k;
        head[h] = nodecnt;
        p = nodecnt;
    }
    return p;
}

inline void writeln(uint64 x) {
    static char buf[20];
    int top = 0;

    if (!x) {
        putchar('0');
        return;
    }
    while (x) {
        top++;
        buf[top] = (x % 10) | 48;
        x /= 10;
    }
    while (top) {
        putchar(buf[top]);
        top--;
    }
    putchar('\n');
}

inline uint64 randint() {
    static uint64 seed;
    return seed = seed * 2333 + 998244353;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("f.in", "r", stdin);
    freopen("f.out", "w", stdout);
#endif

    int i, j, p;
    uint64 ans = 0;

    scanf("%d %d", &n, &m);
    for (i = 1; i <= n; i++) {
        scanf("%d %d", &l[i], &r[i]);
        a[i] = randint();

        f[r[i]] ^= a[i];
        g[l[i]] ^= a[i];
        s[l[i]] ^= a[i];
        s[r[i] + 1] ^= a[i];
        allsum ^= a[i];
    }

    for (i = 1; i <= m; i++)
        f[i] ^= f[i - 1];
    for (i = m; i; i--)
        g[i] ^= g[i + 1];
    for (i = 3; i <= m; i++)
        s[i] ^= s[i - 2];

    for (i = 1; i <= m; i++) {
        p = find(f[i - 1] ^ s[i - 1], true);
        m0[p]++;
        m1[p] += i;
        p = find(s[i] ^ g[i + 1] ^ allsum, false);
        ans += (i + 1) * m0[p] - m1[p];
    }

    memset(s, 0, sizeof(s));
    for (i = 1; i <= n; i++) {
        s[l[i]]++;
        s[r[i] + 1]--;
    }
    for (i = 1; i <= m; i++)
        s[i] += s[i - 1];

    for (i = j = 1; i <= m; i++) {
        while (j <= m and !s[j])
            j++;
        if (i < j)
            ans -= (j - i) * (1llu + j - i) / 2;
        else
            j = i + 1;
    }
    writeln(ans);
    return 0;
}
