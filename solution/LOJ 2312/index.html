LOJ #2312. 「HAOI2017」八纵八横

内存限制：256 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

Anihc 国有 $n$ 个城市，这 $n$ 个城市从 $1\sim n$ 编号，$1$ 号城市为首都。城市间初始时有 $m$ 条高速公路，每条高速公路都有一个非负整数的经济影响因子，每条高速公路的两端都是城市（可能两端是同一个城市)，保证任意两个城市都可以通过高速公路互达。

Anihc 国正在筹划「八纵八横」的高铁建设计划，计划要修建一些高速铁路，每条高速铁路两端也都是城市（可能两端是同一个城市)，也都有一个非负整数的经济影响因子。国家还计划在「八纵八横」计划建成之后，将「一带一路」扩展为「一带一路一环」，增加「内陆城市经济环」即选择一条从首都出发沿若一系列高铁与高速公路走的路径，每条高铁或高速公路可以经过多次，每座城市也可以经过多次，最后路径又在首都结束。令「内陆城市经济环」的 GDP 为依次将这条路径上所经过的高铁或高速公路的经济影响因子异或起来（一条路经过多次则会被计算多次）。

现在 Anihc 在会议上讨论「八纵八横」的建设计划方案，他们会不断地修改计划方案，希望你能实时反馈对于当前的「八纵八横」的建设计划的方案「内陆城市经济环」的最大是多少。

初始时，八纵八横计划中不包含任何—条高铁，有以下三种操作：

Add x y z

在计划中给在城市 $x$ 和城市 $y$ 之间建设一条高铁，其经济影响因子为 $z$ ，如果这是第 $k$ 个 Add 操作，则将这条高铁命名为 $k$ 号高铁。

Cancel k

将计划中的 $k$ 号高铁取消掉，保证此时 $k$ 号高铁一定存在。

Change k z

表示将第 $k$ 号高铁的经济影响因子更改为 $z$ ，保证此时 $k$ 号高铁一定存在。

输入格式

第一行三个整数 $n$ , $m$ , $P$ ，表示城市个数，高速公路条数，操作个数。

接下来 $m$ 行，每行三个整数表示高速公路的信息。

接下来 $P$ 行，每行为一个操作。

注意：输入的所有经济影响因子都将以二进制的形式从高位到低位给出。

输出格式

第一行一个整数，表示如果不修建任何高铁，「内陆城市经济环」的 GDP 最大值。

接下 $Q$ 行，每行一个整数表示进行了对应的每一个操作之后，对于当前的计划「内陆城市经济环」的 GDP 最大值。

注意：输出的答案也要以二进制的形式从高位到低位给出。

样例

样例输入

4 4 3
1 2 1110
1 3 10
2 4 1110
2 3 100
Add 3 4 11
Change 1 101
Cancel 1

样例输出

1000
1001
1111
1000

数据范围与提示

令所有的经济因子二进制表示的最多位数为 $\text{len}$ 。数据满足以下表格：

数据点 $n$ $m$ $Q$ $\text{len}$ 备注
$1$ $\le 5$ $\le 8$ $0$ $\le 31$ 无
$2$ $\le 100$ $= n+1$ $0$ $\le 100$ 无
$3$ $\le 100$ $\le 100$ $0$ $\le 100$ 无
$4$ $\le 500$ $\le 500$ $0$ $\le 1000$ 无
$5$ $\le 100$ $\le 100$ $\le 100$ $\le 200$ 只存在 Add 搡作
$6$ $\le 500$ $\le 500$ $\le 200$ $\le 1000$ 只存在 Add 搡作
$7$ $\le 100$ $\le 100$ $\le 1000$ $\le 200$ 无
$8$ $\le 500$ $\le 500$ $\le 1000$ $\le 1000$ 无
$9$ $\le 500$ $\le 500$ $\le 1000$ $\le 1000$ 无
$10$ $\le 500$ $\le 500$ $\le 1000$ $\le 1000$ 无

对于所有的数据保证：$n,m\le 500,Q,\text{len}\le 1000,1\le x,y\le n$。且 Add 操作不超过 $550$ 个。两个城市之间可能有多条高速公路或高铁，高速公路或高铁的两端可能是同一个城市（即有重边，有自环）。

题解

直接线段树分治套线性基即可。

参照 LOJ 121 「离线可过」动态图连通性（原题 / 题解）的方法，把连边用线段树分治解决。

然后参照 LGOJ P4151 「WC2011」最大 XOR 和路径（原题 / 题解）的方法，用线性基求出最大路径异或值。

可以用 Bitset 优化减小常数。

时间复杂度：$O((n + Q \log Q) \frac{len^2}{64})$。

代码

#include <algorithm>
#include <bitset>
#include <cstdio>
#include <cstring>

const int MAXN = 1000 + 10;
const int MAXLOG = 12;
const int MAXSIZE = MAXN * MAXLOG << 3;

const int BITSIZE = (MAXN - 1) >> 6;

typedef unsigned long long Bitset[BITSIZE + 3];

// Global vars
int n, m, q;

// Graph
int to[MAXN << 1];
Bitset wdis[MAXN << 1];
int pre[MAXSIZE], head[MAXN << 2], wcnt;

// Segtree Add-ons
Bitset segz[MAXSIZE];

// Graph dis
Bitset dis[MAXN];

// Railways
int addx[MAXN], addy[MAXN], addt[MAXN], addcnt;
Bitset addz[MAXN];

// Linear base (Used for segtree)
Bitset base[MAXLOG][MAXN];

// Linear base
inline void reset(Bitset x) { memset(x, 0, sizeof(Bitset)); }
inline void set(Bitset x, int pos) { x[pos >> 6] |= 1ull << (pos & 63); }
inline int size(Bitset x) {
    int i, j;
    i = BITSIZE;
    while (~i and !x[i])
        i--;
    if (~i) {
        j = 63;
        while (!(x[i] >> j))
            j--;
        return (i << 6) | j;
    }
    else
        return -1;
}
inline bool get(Bitset x, int pos) { return x[pos >> 6] >> (pos & 63) & 1; }
inline bool none(Bitset x) { return !(~size(x)); }
inline void xoreq(Bitset x, Bitset y) {
    int i;
    for (i = BITSIZE; ~i; i--)
        x[i] ^= y[i];
}
inline void assign(Bitset x, Bitset y) { memcpy(x, y, sizeof(Bitset)); }
inline bool greater(Bitset x, Bitset y) {
    int i;
    for (i = BITSIZE; ~i; i--)
        if (x[i] > y[i])
            return true;
        else if (x[i] < y[i])
            return false;
    return false;
}

inline void read(Bitset x) {
    static char buf[MAXN];
    char ch;
    int top = 1, i;

    reset(x);

    do
        ch = getchar();
    while (ch < '0');
    while (ch >= '0') {
        top++;
        buf[top] = ch;
        ch = getchar();
    }
    for (i = 0; i < top; i++)
        if (buf[top - i] & 1)
            set(x, i);
}
inline void writeln(Bitset x) {
    int i;

    for (i = size(x); ~i; i--)
        putchar(get(x, i) | 48);
    putchar('\n');
}
inline void ins(Bitset base[], Bitset x) {
    int i;

    for (i = MAXN - 1; ~i; i--) {
        if (get(x, i)) {
            if (none(base[i])) {
                assign(base[i], x);
                return;
            }
            else
                xoreq(x, base[i]);
        }
    }
}
inline void query(Bitset base[], Bitset res) {
    static Bitset temp;
    int i;

    reset(res);
    for (i = MAXN - 1; ~i; i--) {
        assign(temp, res);
        xoreq(temp, base[i]);
        if (greater(temp, res))
            assign(res, temp);
    }
}

// Graph
inline void addedge(int u, int v, Bitset w) {
    wcnt++;
    pre[wcnt] = head[u];
    head[u] = wcnt;
    to[wcnt] = v;
    assign(wdis[wcnt], w);
}
inline void add2edge(int u, int v, Bitset w) {
    addedge(u, v, w);
    addedge(v, u, w);
}

#define v to[i]
#define w wdis[i]
void predfs(int u) {
    static Bitset temp;
    static bool vis[MAXN];
    int i;

    vis[u] = true;
    for (i = head[u]; i; i = pre[i]) {
        if (vis[v]) {
            assign(temp, dis[u]);
            xoreq(temp, w);
            xoreq(temp, dis[v]);
            ins(base[1], temp);
        }
        else {
            assign(dis[v], dis[u]);
            xoreq(dis[v], w);
            predfs(v);
        }
    }
}
#undef v
#undef w

// Segtree
#define lp (p << 1)
#define rp (p << 1 | 1)
#define mid ((l + r) >> 1)
#define lson lp, l, mid
#define rson rp, mid + 1, r
inline void addnode(int p, int x, int y, Bitset z) {
    wcnt++;
    pre[wcnt] = head[p];
    head[p] = wcnt;
    assign(segz[wcnt], z);
    xoreq(segz[wcnt], dis[x]);
    xoreq(segz[wcnt], dis[y]);
}
void update(int p, int l, int r, int a, int b, int x, int y, Bitset z) {
    if (a <= l and r <= b) {
        addnode(p, x, y, z);
        return;
    }
    if (a <= mid)
        update(lson, a, b, x, y, z);
    if (b > mid)
        update(rson, a, b, x, y, z);
}
void solve(int dep, int p, int l, int r) {
    static Bitset temp;
    int i;

    for (i = head[p]; i; i = pre[i])
        ins(base[dep], segz[i]);
    if (l == r) {
        query(base[dep], temp);
        writeln(temp);
        return;
    }
    else {
        memcpy(base[dep + 1], base[dep], sizeof(base[dep]));
        solve(dep + 1, lson);
        memcpy(base[dep + 1], base[dep], sizeof(base[dep]));
        solve(dep + 1, rson);
    }
}
#undef lp
#undef rp
#undef mid
#undef lson
#undef rson

int main() {
#ifndef ONLINE_JUDGE
    freopen("2312.in", "r", stdin);
    freopen("2312.out", "w", stdout);
#endif

    static Bitset w;
    static char opt[10];

    int i;
    int u, v;

    scanf("%d %d %d", &n, &m, &q);
    for (i = 1; i <= m; i++) {
        scanf("%d %d", &u, &v);
        read(w);
        add2edge(u, v, w);
    }

    predfs(1);
    query(base[1], w);
    writeln(w);

    if (!q)
        return 0;

    memset(head, 0, sizeof(head));
    wcnt = 0;
    for (i = 1; i <= q; i++) {
        scanf("%s", opt);
        if (opt[1] == 'd') { // Add
            addcnt++;
            scanf("%d %d", &addx[addcnt], &addy[addcnt]);
            read(addz[addcnt]);
            addt[addcnt] = i;
        }
        else if (opt[1] == 'a') { // Cancel
            scanf("%d", &u);
            update(1, 1, q, addt[u], i - 1, addx[u], addy[u], addz[u]);
            addt[u] = 0;
        }
        else { // Change
            scanf("%d", &u);
            read(w);
            update(1, 1, q, addt[u], i - 1, addx[u], addy[u], addz[u]);
            addt[u] = i;
            assign(addz[u], w);
        }
    }
    for (i = 1; i <= addcnt; i++)
        if (addt[i])
            update(1, 1, q, addt[i], q, addx[i], addy[i], addz[i]);

    solve(1, 1, 1, q);

    return 0;
}
