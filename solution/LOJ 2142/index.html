题目描述

Informatik verbindet dich und mich.

信息将你我连结。

B 君希望以维护一个长度为 $n$ 的数组，这个数组的下标为从 $1$ 到 $n$ 的正整数。

一共有 $m$ 个操作，可以分为两种：

  • $0 \ l \ r$：表示将第 $l$ 个到第 $r$ 个数 $(a_l, a_{l+1}, \dots, a_r)$ 中的每一个数 $a_i$ 替换为 $c^{a_i}$，即 $c$ 的 $a_i$ 次方，其中 $c$ 是输入的一个常数，也就是执行赋值 $$a_i \leftarrow c^{a_i}$$
  • $1 \ l \ r$：求第 $l$ 个到第 $r$ 个数的和，也就是输出： $$\sum_{i = l}^r a_i$$

因为这个结果可能会很大，所以你只需要输出结果 $\mathrel{\mathrm{mod}} p$ 的值即可。

输入格式

第一行有四个整数 $n, m, p, c$，所有整数含义见问题描述。

接下来一行 $n$ 个整数，表示 $a$ 数组的初始值。

接下来 $m$ 行，每行三个整数，其中第一个整数表示了操作的类型。

  • 如果是 $0$ 的话，表示这是一个修改操作，操作的参数为 $l, r$。
  • 如果是 $1$ 的话，表示这是一个询问操作，操作的参数为 $l, r$。

输出格式

对于每个询问操作，输出一行，包括一个整数表示答案 $\mathrel{\mathrm{mod}} p$ 的值。

题解

扩展欧拉定理的运用。

根据扩展欧拉定理，这个数修改次数不会超过 $2 \log p$ 次。

使用类似维护区间开方的方法，暴力修改。

这里使用了一个快速幂的技巧：假设我们要求 $a^p \bmod P$。

令 $m = \lceil \sqrt P \rceil$，把 $a^p$ 拆分为 $a^{im+j}$，那么 $O(\sqrt P)$ 预处理出所有 $a^{im} \: (0 \leq i \leq m)$ 和 $a^j \: (0 \leq j < m)$ 就可以 $O(1)$ 求出快速幂了。

这个思想和 bsgs 是差不多的。

注意处理 $c = 1$ 的特殊情况。

至于如何知道指数是否超过 $\varphi(p)$，可以考虑预处理出 $lima[i][j]$ 表示令 $c^{c^{\ldots^{a}}} \geq \varphi(\varphi(\ldots\varphi(p)))$ 成立的最小非负整数 $a$ 的值，在这里有 $(i-j)$ 个 $c$ 和 $j$ 个 $\varphi$。

$lima$ 可以这样计算： $$ \begin{aligned} c^{c^{c^a}} &\geq n \\ c^{c^a} &\geq \lceil \log_c n \rceil \\ c^a &\geq \lceil \log_c \lceil \log_c n \rceil \rceil\\ a &\geq \lceil \log_c \lceil \log_c \lceil \log_c n \rceil \rceil \rceil\\ \end{aligned} $$

具体可以参照代码。

时间复杂度：$O(n \log n \log^2 p)$。

当然，在 $m \leq 50000$ 的限制内远远达不到这个时间复杂度。

代码

#include <algorithm>
#include <cmath>
#include <cstdio>

#define lp (p << 1)
#define rp (p << 1 | 1)
#define mid ((l + r) >> 1)
#define lson lp, l, mid
#define rson rp, mid + 1, r

typedef long long int64;

const int MAXN = 50000 + 10;
const int MAXT = 40;
const int MAXB = 1 << 15;

int n, m, c, MOD;
int sum[MAXN << 2], mark[MAXN << 2];
int a[MAXN];

int phif[MAXT + 2];
int lima[MAXT + 2][MAXT + 2];
int cpow1[MAXT + 2][MAXB];
int cpow2[MAXT + 2][MAXB];

// Math
int log(int n, int base) { // ceil log
    int prod = 1, ans = 0;

    if (n == 1)
        return 0;
    while ((int64)prod * base <= n) {
        ans++;
        prod *= base;
    }
    return ans + (prod < n);
}
int phi(int n) {
    int res = n, i;

    for (i = 2; i * i <= n; i++) {
        if (!(n % i)) {
            res = res / i * (i - 1);
            while (!(n % i))
                n /= i;
        }
    }
    if (n - 1)
        res = res / n * (n - 1);
    return res;
}
void init() {
    int i, j, k, cp;

    phif[0] = MOD;
    for (i = 1; i <= MAXT; i++)
        phif[i] = phi(phif[i - 1]);

    for (i = 0; i <= MAXT; i++) {
        cpow1[i][0] = 1;
        for (j = 1; j < MAXB; j++)
            cpow1[i][j] = (int64)cpow1[i][j - 1] * c % phif[i];

        cp = (int64)cpow1[i][MAXB - 1] * c % phif[i];
        cpow2[i][0] = 1;
        for (j = 1; j < MAXB; j++)
            cpow2[i][j] = (int64)cpow2[i][j - 1] * cp % phif[i];
    }

    for (i = 1; i <= MAXT; i++) {
        for (j = i - 1; ~j; j--) {
            lima[i][j] = phif[j];
            for (k = 1; k <= i - j and lima[i][j]; k++)
                lima[i][j] = log(lima[i][j], c);
        }
    }
}
inline int qpow(int p, int i) { return (int64)cpow1[i][p % MAXB] * cpow2[i][p / MAXB] % phif[i]; }

// Segtree
inline void updatenode(int p, int no) {
    int i, x, y;

    if (c == 1) {
        sum[p] = bool(MOD);
        mark[p] = MAXT;
    }
    else {
        mark[p]++;
        sum[p] = a[no] % phif[mark[p]];
        for (i = mark[p] - 1; ~i; i--) {
            if (a[no] >= lima[mark[p]][i])
                sum[p] = qpow(sum[p] + phif[i + 1], i);
            else
                sum[p] = qpow(sum[p], i);
        }
    }
}
inline void up(int p) {
    sum[p] = (sum[lp] + sum[rp]) % MOD;
    mark[p] = std::min(mark[lp], mark[rp]);
}
void build(int p, int l, int r) {
    if (l ^ r) {
        build(lson);
        build(rson);
        up(p);
    }
    else {
        sum[p] = a[l] % MOD;
        mark[p] = 0;
    }
}
void update(int p, int l, int r, int a, int b) {
    if (mark[p] >= MAXT)
        return;
    if (l == r) {
        updatenode(p, l);
        return;
    }
    if (a <= mid)
        update(lson, a, b);
    if (b> mid)
        update(rson, a, b);
    up(p);
}
int query(int p, int l, int r, int a, int b) {
    int res = 0;
    if (a <= l and r <= b)
        return sum[p];
    if (a <= mid)
        res = query(lson, a, b);
    if (b> mid)
        res += query(rson, a, b);
    return res % MOD;
}

int main() {
#ifndef ONLINE_JUDGE

    freopen("2142.out", "w", stdout);
#endif

    int i, opt, l, r;

    scanf("%d %d %d %d", &n, &m, &MOD, &c);

    if (c> 1)
        init();

    for (i = 1; i <= n; i++)
        scanf("%d", &a[i]);
    build(1, 1, n);

    for (i = 1; i <= m; i++) {
        scanf("%d %d %d", &opt, &l, &r);
        if (opt == 0)
            update(1, 1, n, l, r);
        else
            printf("%d\n", query(1, 1, n, l, r));
    }
    return 0;
}
