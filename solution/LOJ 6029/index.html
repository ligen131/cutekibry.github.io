LOJ #6029. 「雅礼集训 2017 Day1」市场

题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

从前有一个贸易市场，在一位执政官到来之前都是非常繁荣的，自从他来了之后，发布了一系列奇怪的政令，导致贸易市场的衰落。

有 $n$ 个商贩，从 $0 \sim n - 1$ 编号，每个商贩的商品有一个价格 $a_i$，有两种政令：

  1. $l, r, c$，对于 $i \in [l, r], a_i \leftarrow a_i + c$
  2. $l, r, d$，对于 $i \in [l, r], a_i \leftarrow \lfloor {a_i}/{d} \rfloor$

现在有一个外乡的旅客想要了解贸易市场的信息，有两种询问方式：

  • 给定 $l, r$，求 $\min_{i \in [l, r]} a_i$
  • 给定 $l, r$，求 $\sum_{i\in [l, r]} a_i$

输入格式

第一行为两个空格隔开的整数 $n, q$ 分别表示商贩个数和政令 + 询问个数。

第二行包含 $n$ 个由空格隔开的整数 $a_0 \sim a_{n - 1}$

接下来 $q$ 行，每行表示一个操作，第一个数表示操作编号 $1 \sim 4$，接下来的输入和问题描述一致。

输出格式

对于每个 3、4 操作，输出询问答案。

样例输入

-5 -4 -3 -2 -1 0 1 2 3 4
1 0 4 1
1 5 9 1

数据范围与提示

对于 $30\%$ 的数据，$n, q \leq 10 ^ 3$；

对于 $60\%$ 的数据，保证数据随机；

对于 $100\%$ 的数据，$1 \leq n, q \leq 10 ^ 5, 0 \leq l \leq r \leq n - 1, c \in [-10 ^ {4}, 10 ^ 4], d \in [2, 10 ^ 9]$

题解

注意到要维护区间除法，且除法 $d \leq 10^9$ 但 $|c|$ 却不超过 $10^4$，这让我们考虑使用势能线段树。

显然对区间下溢除法是很难做的，不如让我们考虑把区间除改为我们熟悉的操作——区间减。

显然，对于一堆元素的集合 $S$，若满足

$$x - \lfloor \frac xd \rfloor = y - \lfloor \frac yd \rfloor \quad (x, y \in S)$$

则可以把区间除变为区间减。

可以证明，只需要判断区间最大值和区间最小值减小的数是否相同即可。

题目则变为维护区间加，区间减，区间最大值，区间最小值。

时间复杂度：$O(\text{能过})$。

代码

#include <algorithm>
#include <cstdio>

#define lp (p << 1)
#define rp (p << 1 | 1)
#define mid ((l + r) >> 1)
#define lson lp, l, mid
#define rson rp, mid + 1, r

typedef long long int64;

const int MAXN = 100000 + 10;

int n, m;
int a[MAXN];
int64 sum[MAXN << 2], minv[MAXN << 2], maxv[MAXN << 2], mark[MAXN << 2];

inline int64 floor(int64 x, int y) { return (x < 0) ? ((x - y + 1) / y) : (x / y); }

inline void addnode(int p, int l, int r, int64 x) {
    sum[p] += (r - l + 1) * x;
    minv[p] += x;
    maxv[p] += x;
    mark[p] += x;
}
inline void down(int p, int l, int r) {
    if (mark[p]) {
        addnode(lson, mark[p]);
        addnode(rson, mark[p]);
        mark[p] = 0;
    }
}
inline void up(int p) {
    sum[p] = sum[lp] + sum[rp];
    minv[p] = std::min(minv[lp], minv[rp]);
    maxv[p] = std::max(maxv[lp], maxv[rp]);
}
void add(int p, int l, int r, int a, int b, int x) {
    if (a <= l and r <= b) {
        addnode(p, l, r, x);
        return;
    }
    down(p, l, r);
    if (a <= mid)
        add(lson, a, b, x);
    if (b > mid)
        add(rson, a, b, x);
    up(p);
}
void div(int p, int l, int r, int a, int b, int x) {
    if (a <= l and r <= b and minv[p] - floor(minv[p], x) == maxv[p] - floor(maxv[p], x)) {
        addnode(p, l, r, floor(minv[p], x) - minv[p]);
        return;
    }
    down(p, l, r);
    if (a <= mid)
        div(lson, a, b, x);
    if (b > mid)
        div(rson, a, b, x);
    up(p);
}
int64 querysum(int p, int l, int r, int a, int b) {
    int64 res = 0;
    if (a <= l and r <= b)
        return sum[p];
    down(p, l, r);
    if (a <= mid)
        res += querysum(lson, a, b);
    if (b > mid)
        res += querysum(rson, a, b);
    return res;
}
int64 querymin(int p, int l, int r, int a, int b) {
    int64 res = 1e18;
    if (a <= l and r <= b)
        return minv[p];
    down(p, l, r);
    if (a <= mid)
        res = std::min(res, querymin(lson, a, b));
    if (b > mid)
        res = std::min(res, querymin(rson, a, b));
    return res;
}
void build(int p, int l, int r) {
    if (l ^ r) {
        build(lson);
        build(rson);
        up(p);
    }
    else
        sum[p] = minv[p] = maxv[p] = a[l];
}
void midorder(int p, int l, int r) {
    if (l ^ r) {
        down(p, l, r);
        midorder(lson);
        midorder(rson);
    }
    else
        printf("%lld ", sum[p]);
}

    int i, opt, l, r, x;

    scanf("%d %d", &n, &m);
    for (i = 1; i <= n; i++)
        scanf("%d", &a[i]);
    build(1, 1, n);
    // printf("# ");
    // midorder(1, 1, n);
    // putchar('\n');
    for (i = 1; i <= m; i++) {
        scanf("%d %d %d", &opt, &l, &r);
        l++;
        r++;
        if (opt == 1) {
            scanf("%d", &x);
            add(1, 1, n, l, r, x);
        }
        else if (opt == 2) {
            scanf("%d", &x);
            div(1, 1, n, l, r, x);
        }
        else if (opt == 3)
            printf("%lld\n", querymin(1, 1, n, l, r));
        else
            printf("%lld\n", querysum(1, 1, n, l, r));
        // printf("# ");
        // midorder(1, 1, n);
        // putchar('\n');
    }
    return 0;
}
