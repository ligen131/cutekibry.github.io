LOJ #6169. 相似序列

内存限制：256 MiB 时间限制：2000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

给定长度为 $n$ 的整数序列 $A$，你需要回答 $q$ 个询问。询问给定两个子串 $[a, b]$ 和 $[c, d]$，要求你判断两个子串是否相似。

如果两个序列在各自排序后，除最多一个位置外，其他所有位置上的元素对应相等，那么我们认为两个序列相似。

请注意，给定的子串可以有公共部分，但不会互相影响。

输入格式

输入的第一行包含一个整数 $T$ ，代表测试数据的组数。接下来是 $T$ 组数据。

每组数据的第一行包含两个整数 $n$ 和 $q$，分别代表序列长度和询问数。

接下来一行包含 $n$ 个由空格分隔的整数 $A_1, A_2, \dots , A_n$。

接下来 Q 行，每行描述一个询问。询问以四个整数 $a, b, c, d$ 表示，其中 $a$ 和 $b$ 为第一个子串的左右端点，$c$ 和 $d$ 为第二个子串的左右端点。

输出格式

对于每个询问，判断两个子串是否相似，并相应输出一行 YES 或者 NO 。

样例

样例输入

1
6 3
1 3 4 2 3 4
1 3 4 6
1 2 5 6
3 5 2 4

样例输出

YES
NO
YES

样例解释

在第一个询问中，第一个子串在排序后为 $[1, 3, 4]$，第二个子串在排序后为 $[2, 3, 4]$。除第 $1$ 个位置上的元素外，两个子串对应相等，故相似。

在第二个询问中，第一个子串在排序后为 $[1, 3]$，第二个子串在排序后为 $[3, 4]$。两个位置上的元素均不相等，故不相似。

在第三个询问中，第一个子串在排序后为 $[2, 3, 4]$，第二个子串在排序后为 $[2, 3, 4]$。两个子串对应位置完全相等，故相似。

数据范围与提示

  • 所有数据满足：
    • $ 1 \leq a \leq b \leq n $
    • $ 1 \leq c \leq d \leq n $
    • $ b − a = d − c $
  • 子任务 1（10 分）
    • $ 1 \leq T \leq 3 $
    • $ 1 \leq n,q \leq 10 ^ 3 $
    • $ 1 \leq A[i] \leq 10 ^ 3 $
  • 子任务 2（20 分）
    • $ 1 \leq T \leq 3 $
    • $ 1 \leq n \leq 10 ^ 5 $
    • $ 1 \leq q \leq 10 ^ 4 $
    • $ 1 \leq A_i \leq 10 ^ 5 $
  • 子任务 3（70 分）
    • $ 1 \leq T \leq 3 $
    • $ 1 \leq n,q \leq 10 ^ 5 $
    • $ 1 \leq A_i \leq 10 ^ 5 $

题解

可以先考虑区间线段树维护 Hash。也就是说，考虑某种 Hash 的方法，使得我们可以用区间线段树求出某个区间的 Hash 值，然后判断两个区间的 Hash 值相同即可。

容易注意到本题的 Hash 必须满足交换律和结合律，也就是对于任意序列 $A, B$，有 $\text{hash}(A) * \text{hash}(B) = \text{hash}(A + B)$（$*$ 是一个运算符）。

显然，一般的 $k$ 进制 Hash、异或 Hash 都不满足该条件。

那就用加法 Hash 吧。

直接给每个 $A_i$ 赋值一个随机值 $val_i \: (val_i \in [0, 2^{64} - 1]$，然后定义某个区间的 Hash 值为区间所有元素的 $val_i$ 之和。

如果两个区间的 Hash 值相等，则可以认为这两个区间重排后完全相同。

由于 $val_i$ 在 $[0, 2^{64} - 1]$ 之间等概率随机，每种重排后不同的区间的 Hash 值也在 $[0, 2^{64} - 1]$ 之间等概率随机。因此，该方法出错的概率为 $\frac {1}{2^{64}}$。

构建前缀和权值线段树，在线段树上二分不同的位置即可。

int divquery(int l, int r, int p1, int p2, int p3, int p4) {
    if (l == r)
        return l;
    else if ((sum[lson[p2]] - sum[lson[p1]]) ^ (sum[lson[p4]] - sum[lson[p3]]))
        return divquery(l, mid, lson[p1], lson[p2], lson[p3], lson[p4], inp1);
    else
        return divquery(mid + 1, r, rson[p1], rson[p2], rson[p3], rson[p4], inp1);
}

实现细节留作读者思考。

时间复杂度：$O(n + q \log A_i)$。

代码

#include <algorithm>
#include <cstdio>
#include <cstring>

typedef unsigned long long uint64;

#ifdef ONLINE_JUDGE
const int MAXN = 100000 + 10;
#else
const int MAXN = 100 + 10;
#endif

const int MAXSIZE = MAXN * 40;

int n, m, q;

int a[MAXN];

uint64 val[MAXN];

int rt[MAXN];
uint64 sum[MAXSIZE];
int size[MAXSIZE];
int lson[MAXSIZE], rson[MAXSIZE];
int nodecnt;

#define mid ((l + r) >> 1)
#define lc lson[p], l, mid
#define rc rson[p], mid + 1, r
void ins(int &p, int l, int r, int x) {
    nodecnt++;
    sum[nodecnt] = sum[p] + val[x];
    size[nodecnt] = size[p] + 1;
    lson[nodecnt] = lson[p];
    rson[nodecnt] = rson[p];
    p = nodecnt;
    if (l == r)
        return;
    else if (x <= mid)
        ins(lc, x);
    else
        ins(rc, x);
}
uint64 query(int p, int l, int r, int a, int b) {
    if (!sum[p] or (a <= l and r <= b) or a > b)
        return sum[p];
    else if (a <= mid and b > mid)
        return query(lc, a, b) + query(rc, a, b);
    else if (a <= mid)
        return query(lc, a, b);
    else
        return query(rc, a, b);
}
int divquery(int l, int r, int p1, int p2, int p3, int p4, bool &inp1) {
    if (l == r)
        inp1 = sum[p2] - sum[p1] - sum[p4] + sum[p3] + val[l];

    if (l == r)
        return l;
    else if ((sum[lson[p2]] - sum[lson[p1]]) ^ (sum[lson[p4]] - sum[lson[p3]]))
        return divquery(l, mid, lson[p1], lson[p2], lson[p3], lson[p4], inp1);
    else
        return divquery(mid + 1, r, rson[p1], rson[p2], rson[p3], rson[p4], inp1);
}
int leq(int l, int r, int p1, int p2, int x) {
    if (!p1 and !p2)
        return 0;
    else if (x <= mid)
        return leq(l, mid, lson[p1], lson[p2], x);
    else
        return size[lson[p2]] - size[lson[p1]] + leq(mid + 1, r, rson[p1], rson[p2], x);
}
int kth(int l, int r, int p1, int p2, int k) {
    if (k > size[p2] - size[p1])
        return MAXN - 1;
    else if (l == r)
        return l;
    else if (k <= size[lson[p2]] - size[lson[p1]])
        return kth(l, mid, lson[p1], lson[p2], k);
    else
        return kth(mid + 1, r, rson[p1], rson[p2], k - size[lson[p2]] + size[lson[p1]]);
}
int after(int p1, int p2, int x) { return kth(1, m, p1, p2, leq(1, m, p1, p2, x) + 1); }
#undef mid
#undef lc
#undef rc

int main() {
#ifndef ONLINE_JUDGE
    freopen("6169.in", "r", stdin);
    freopen("6169.out", "w", stdout);
#endif

    int T;
    int i, x1, x2;
    int l1, r1, l2, r2;
    uint64 x;
    bool inp1;

    val[1] = 2773467721;
    for (i = 2; i < MAXN; i++)
        val[i] = val[i - 1] * 10007 + 998244353;

    scanf("%d", &T);
    while (T) {
        nodecnt = 0;
        memset(rt, 0, sizeof(rt));
        memset(sum, 0, sizeof(sum));
        memset(lson, 0, sizeof(lson));
        memset(rson, 0, sizeof(rson));
        memset(size, 0, sizeof(size));

        scanf("%d %d", &n, &q);
        m = 0;
        for (i = 1; i <= n; i++) {
            scanf("%d", &a[i]);
            m = std::max(m, a[i]);
        }

        for (i = 1; i <= n; i++) {
            rt[i] = rt[i - 1];
            ins(rt[i], 1, m, a[i]);
        }

        for (i = 1; i <= q; i++) {
            scanf("%d %d %d %d", &l1, &r1, &l2, &r2);
            if ((r1 - l1) ^ (r2 - l2))
                printf("NO\n");
            else {
                x = sum[rt[r1]] - sum[rt[l1 - 1]] - sum[rt[r2]] + sum[rt[l2 - 1]];
                if (!x) {
                    printf("YES\n");
                    continue;
                }

                x1 = divquery(1, m, rt[l1 - 1], rt[r1], rt[l2 - 1], rt[r2], inp1);
                if (inp1)
                    x2 = after(rt[l2 - 1], rt[r2], x1);
                else
                    x2 = after(rt[l1 - 1], rt[r1], x1);

                if (x == val[x1] - val[x2] or x == val[x2] - val[x1])
                    printf("YES\n");
                else
                    printf("NO\n");
            }
        }

        T--;
    }
    return 0;
}
