LOJ #3110. 「SDOI2019」快速查询

内存限制：512 MiB 时间限制：2000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

给定一个长度为 $n$ 的整数数列，里面的元素依次编号为 $a_1,a_2,a_3,\cdots,a_n$。初始的时候，所有元素都为零。现在按照时间顺序提供了若干次关于这个数列的修改或询问，每一次修改或询问一定为以下六种情况之一：

  • $1\ i\ \text{val}$：将 $a_i$ 赋值为给定整数 $\text{val}$；
  • $2\ \text{val}$：将所有元素同时加上 $\text{val}$；
  • $3\ \text{val}$：将所有元素同时乘上 $\text{val}$；
  • $4\ \text{val}$：将所有元素同时赋值为 $\text{val}$；
  • $5\ i$：询问第 $i$ 个元素 $a_i$ 现在的值是多少；
  • $6$：询问现在所有元素的和。

输入格式

为了避免读入太大，输入文件采取如下的形式。

第一行给定整数 $n$，表示给定数列长度为 $n$。

第二行给定整数 $q$，并且之后的 $q$ 行，每一行提供一个修改或询问，输入的格式与题目所述一致，请参见样例。

我们称上述给定的修改或询问为标准操作。

之后给定一个整数 $t$，并且之后的 $t$ 行每行给定两个正整数 $a_i$ 和 $b_i$，这里的下标 $i$ 依次记为 $1$ 到 $t$。

你需要对初始值全为零的长度为 $n$ 的序列做总计 $t\times q$ 次操作。

其中第 $\Big((i-1)q+j\Big)$ 次操作形如第 $\Big((a_i + j b_i) \bmod{q} + 1\Big)$ 个给定的标准操作（$1\le i\le t$ 且 $1\le j\le q$）。

输出格式

输出一个整数，表示所有询问答案的累计和。

因为答案可能很大，只要求输出其结果关于 $p=10^7+19$ 取模后的值。

注意：若最终的累计和 $\text{ans}$ 小于零，你应该输出 $\big((\text{ans} \bmod{p})+p\big)\bmod{p})$。

数据范围与提示

子任务 $1$（$50$ 分）：$1\le n\le 500000$，$1\le q\le 10^5$ 且 $1\le t\le 5$，所有在输入中出现的 $\text{val}$ 满足 $-10^9\le \text{val}\le 10^9$，所有 $a_i$ 和 $b_i$ 满足 $0\le a_i,b_i\le 10^9$。

子任务 $2$（$50$ 分）：$1\le n\le 10^9$，$1\le q\le 10^5$ 且 $1\le t\le 100$，所有在输入中出现的 $\text{val}$ 满足$-10^9\le \text{val}\le 10^9$，所有 $a_i$ 和 $b_i$ 满足 $0\le a_i,b_i\le 10^9$。

题解

我会线段树！

然而操作数 $t \times q \leq 10^7$，$n \leq 10^9$，很容易超时。

容易注意到，除了单点赋值，其他修改操作都是全局的。

考虑全局维护标记 $k, b$，代表第 $i$ 个元素为 $ka_i + b$。然后对于赋值 $a_i$ 为 $x$ 的操作，我们令其赋值为 $ky + b = x$ 的 $y$ 即可（因为要适应全局标记的影响）。

显然，由于 $p = 10^7 + 19$ 是一个质数，所以当 $k \bmod p \neq 0$ 时必定存在这样的 $y$。

$$y \equiv \frac {x - b}k \equiv (x - b)k^{p - 2} \pmod p$$

当然，$k$ 可能为 $0$。在这种情况下，维护多一个全局消为 $0$ 的标记即可。

用 Hash 表维护。

说实话，这道题真的简单过头了。

时间复杂度：$O(tq)$。

代码

#include <cstdio>

typedef long long int64;

const int HMOD = 10000079;
const int MOD = 10000019;

#ifdef ONLINE_JUDGE
const int MAXQ = 100000 + 10;
#else
const int MAXQ = 1000 + 10;
#endif

int n;
int head[HMOD];
int pre[MAXQ], key[MAXQ], nodecnt;
int val[MAXQ], last[MAXQ];
int m1 = 1, m2, t0 = -1, timecnt, sum;

inline int gethash(int x) { return x % HMOD; }
inline int find(int x, bool create = false) {
    int k = gethash(x), p = head[k];
    while (key[p] ^ x and p)
        p = pre[p];
    if (!p and create) {
        nodecnt++;
        pre[nodecnt] = head[k];
        key[nodecnt] = x;
        head[k] = nodecnt;
        p = nodecnt;
    }
    return p;
}

inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
inline int qpow(int64 b, int p, int mod = MOD) {
    int res = 1;

    while (p) {
        if (p & 1)
            res = res * b % mod;
        b = b * b % mod;
        p >>= 1;
    }
    return res;
}

inline void updall(int x1, int x2) {
    m1 = 1ll * m1 * x1 % MOD;
    m2 = (1ll * m2 * x1 + x2) % MOD;
    if (!m1) {
        m1 = 1;
        sum = 0;
        t0 = timecnt;
    }
}
#define addall(x) updall(1, x)
#define mulall(x) updall(x, 0)
#define fixall(x) updall(0, x)
inline void fix(int p, int x) {
    p = find(p);
    if (last[p] < t0)
        val[p] = 0;
    last[p] = timecnt;

    x = 1ll * qmod(MOD + x - m2) * qpow(m1, MOD - 2) % MOD;
    sum = qmod(sum + qmod(MOD + x - val[p]));
    val[p] = x;
}
inline int query(int p) {
    p = find(p);
    if (last[p] < t0)
        val[p] = 0;
    last[p] = timecnt;

    return (1ll * m1 * val[p] + m2) % MOD;
}
inline int queryall() { return (1ll * m1 * sum + 1ll * m2 * n) % MOD; }

int main() {
#ifndef ONLINE_JUDGE
    freopen("3110.in", "r", stdin);
    freopen("3110.out", "w", stdout);
#endif

    static int opt[MAXQ], x[MAXQ], y[MAXQ];
    int i, j;
    int q, t, pp;
    int a, b;
    int ans = 0;

    scanf("%d %d", &n, &q);
    for (i = 1; i <= q; i++) {
        scanf("%d", &opt[i]);
        if (opt[i] <= 5)
            scanf("%d", &x[i]);
        if (opt[i] == 1) {
            scanf("%d", &y[i]);
            find(x[i], true);
            y[i] = qmod(MOD + y[i] % MOD);
        }
        if (opt[i] >= 2 and opt[i] <= 4)
            x[i] = qmod(MOD + x[i] % MOD);
    }

    scanf("%d", &t);
    for (i = 1; i <= t; i++) {
        scanf("%d %d", &a, &b);
        for (j = 1; j <= q; j++) {
            pp = (a + 1ll * j * b) % q + 1;
            switch (opt[pp]) {
            case 1: {
                fix(x[pp], y[pp]);
                break;
            }
            case 2: {
                addall(x[pp]);
                break;
            }
            case 3: {
                mulall(x[pp]);
                break;
            }
            case 4: {
                fixall(x[pp]);
                break;
            }
            case 5: {
                ans = qmod(ans + query(x[pp]));
                break;
            }
            case 6: {
                ans = qmod(ans + queryall());
                break;
            }
            }
            timecnt++;
        }
    }
    printf("%d\n", ans);
    return 0;
}
