LOJ #3069. 「2019 集训队互测 Day 1」整点计数

内存限制：256 MiB 时间限制：4000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

小 X 的姐姐小 F 是一名 X 国的占星师，而作为附魔师的小 X 则需要在占星的过程中辅助小 F。

今夜，空中的星星排列得格外整齐，如果将整个星空抽象成一个平面直角坐标系的话，在每一个整点处，都恰好有一颗星，小 X 和小 F 所在的星就是原点。

距离小 X 和小 F 所在的星相同的星之间是会相互作用的，如果记 $f(x)$ 表示距离小 X 和小 F 所在的星 $x$ 个单位的星星的个数，那么这一系列的星将共计产生 $f(x)^k$ 点能量。

小 F 观察到，在月圆之夜，距离自己和小 X 所在的星恰好整数个单位的星产生的能量将会变得易于收集。并且，由于技术限制，小 F 暂时只能够收集距离自己和小 X 所在的星不超过 $N$ 个单位的星产生的能量。需要特别注意的是，小 X 和小 F 所在的星并不会产生能量。

小 X 要做的，就是帮助小 F 计算此时能够被收集的能量总量。由于能够收集的能量实在太过庞大，小 X 难以确保自己计算毫无失误，因此他希望你能够告诉他这个数值在对某一个数 $P$ 取模后的结果来验证他计算的正确性。

输入格式

一行三个正整数 $N,k,P$。

输出格式

一行一个整数 $\text{Ans}$，表示能量总量对 $P$ 取模的结果。

样例

样例输入 1

5 1 998244353

样例输出 1

28

样例说明

样例输入中 $k=1$，不难发现此时小 $X$ 需要计算的就是距原点 $5$ 单位内，且至原点距离为整数的整点个数，对 $998244353$ 取模的结果。

形如 $(x,0),(-x,0),(0,x),(0,-x)$ 的符合要求的整点有 $4\times 5=20$ 个；除了以上整点以外，还有 $(3,4),(-3,4),(3,-4),(-3,-4),(4,3),(-4,3),(4,-3),(-4,-3)$ 共 $8$ 个符合要求的整点，总计 $28$ 个。

样例输入 2

500 5 1000000000

样例输出 2

365511168

样例输入 3

142857 1 1000000009

样例输出 3

2481012

样例输入 4

998244353 998244353 998244353

样例输出 4

637246480

题解

根据「圆上的整点」，我们可以设 $g(n) = \frac {f(n)}4 = \sum_{d | n^2} \chi(d)$，那么答案即是

$$4^k \sum_{i = 1}^N g(i)^k$$

两个积性函数的狄利克雷卷积也是一个积性函数，一个积性函数的幂函数也是积性函数，因而 $g^k$ 也是积性函数。

注意到 $N \leq 10^{11}$，同时 $\chi$ 也没有什么好的性质能用到杜教筛中，所以我们只能考虑洲阁筛。

那么对于任意质数 $p$ 和正整数 $k$，有

$$ g(p^k)^K = \begin{cases} (2k + 1)^K, &p \bmod 4 = 1 \\ 1, &p \bmod 4 \neq 1 \\ \end{cases} $$

可以根据这一点完成洲阁筛第一部分的计算。

特别地，当 $k = 1$ 时，有

$$ g(p)^K = \begin{cases} 3^K, &p \bmod 4 = 1 \\ 1, &p \bmod 4 = 3 \\ \end{cases} $$

这相当于求出 $\bmod \: 4$ 分别为 $1, 3$ 的质数个数，可参见 LOJ 6027 「from CommonAnts」质数计数 I（原题 / 题解）。

可以根据这一点完成洲阁筛第二部分的计算。

时间复杂度：$O(\frac {n^{\frac 34}}{\log n})$。

代码

#include <cmath>
#include <cstdio>

typedef long long int64;

// #ifdef ONLINE_JUDGE
const int MAXB = 8000000 + 10;
// #else
// const int MAXB = 1000 + 10;
// #endif

const int MAXLOG = 50 + 10;

int MOD;
int64 n, m;
int B;

// POWK[i] = i ** k
int POWK[MAXLOG];

// floor(n / i)
int64 d[MAXB];
int dn;

// Primes not exceed B
int p[MAXB], pn;

int pgsum[MAXB];              // prefix sum of g(p)
int p1sum[MAXB], p3sum[MAXB]; // prefix sum of p mod 4 == 1 / 3
int g[MAXB];                  // g(p)

// Result of zz_sleve()
int F1;
int F2[MAXB];

inline int min(int64 x, int64 y) { return (x < y) ? x : y; }
inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
inline int qpow(int64 b, int64 p, int mod = MOD) {
    int res = 1;

    while (p) {
        if (p & 1)
            res = res * b % mod;
        b = b * b % mod;
        p >>= 1;
    }
    return res;
}
inline void linear_sleve() {
    static bool notp[MAXB];
    int i, j;
    int64 x, y;

    for (i = 1; i < MAXLOG; i++)
        POWK[i] = qpow(i, m);

    notp[1] = true;
    g[1] = 1;
    for (i = 2; i <= B; i++) {
        if (!notp[i]) {
            pn++;
            p[pn] = i;
            for (j = 3, x = i; x <= B; x *= i, j += 2)
                g[x] = (i & 2) ? 1 : POWK[j];
        }
        for (j = 1; j <= pn and i * p[j] <= B; j++) {
            notp[i * p[j]] = true;
            if (i % p[j])
                g[i * p[j]] = 1ll * g[i] * g[p[j]] % MOD;
            else {
                x = i;
                y = p[j];
                while (!(x % p[j])) {
                    x /= p[j];
                    y *= p[j];
                }
                g[i * p[j]] = 1ll * g[x] * g[y] % MOD;
                break;
            }
        }
    }

    for (i = 2; i <= B; i++) {
        p1sum[i] = p1sum[i - 1];
        p3sum[i] = p3sum[i - 1];
        pgsum[i] = pgsum[i - 1];
        if (!notp[i]) {
            pgsum[i] = qmod(pgsum[i - 1] + g[i]);
            if ((i & 3) == 1)
                p1sum[i] = qmod(p1sum[i - 1] + 1);
            if ((i & 3) == 3)
                p3sum[i] = qmod(p3sum[i - 1] + 1);
        }
    }
}

void zz_sleve() {
    static int last[MAXB];
    static int f1[MAXB], f3[MAXB];
    int i, j, k, pos;
    int64 x;

    for (x = n; x; x = n / (d[dn] + 1)) {
        dn++;
        d[dn] = n / x;
    }

    // Sleve 1
    for (i = 1; i <= dn; i++)
        f1[i] = 1;
    for (i = pn; i; i--) {
        for (j = dn; j; j--) {
            x = d[j] / p[i];
            if (x < p[i])
                break;

            if (i == pn or d[j] < p[i + 1])
                f1[j] = 1;
            else if (d[j] / p[i + 1] < p[i + 1])
                f1[j] = qmod(1 + qmod(MOD + pgsum[min(B, d[j])] - pgsum[p[i]]));

            for (k = 3; x; x /= p[i], k += 2) {
                pos = (x <= B) ? x : (dn - n / x + 1);

                if (x < p[i + 1] or i == pn)
                    f1[pos] = 1;
                else if (x / p[i + 1] < p[i + 1])
                    f1[pos] = qmod(1 + qmod(MOD + pgsum[min(B, x)] - pgsum[p[i]]));

                if ((p[i] & 3) ^ 1)
                    k = 1;
                f1[j] = (f1[j] + 1ll * POWK[k] * f1[pos]) % MOD;
            }
        }
    }
    F1 = f1[dn];

    // Sleve 2
    for (i = 1; i <= dn; i++) {
        f1[i] = ((d[i] + 3) >> 2) % MOD;
        f3[i] = ((d[i] + 1) >> 2) % MOD;
    }
    for (i = 2; i <= pn; i++) {
        for (j = dn; j; j--) {
            x = d[j] / p[i];
            if (x < p[i])
                break;

            pos = (x <= B) ? x : (dn - n / x + 1);
            f1[pos] = qmod(f1[pos] + qmod(MOD + p1sum[p[last[pos]]] - p1sum[p[i - 1]]));
            f3[pos] = qmod(f3[pos] + qmod(MOD + p3sum[p[last[pos]]] - p3sum[p[i - 1]]));
            last[pos] = i - 1;

            if ((p[i] & 3) == 3) {
                f1[j] = qmod(MOD + f1[j] - f3[pos]);
                f3[j] = qmod(MOD + f3[j] - f1[pos]);
            }
            else {
                f1[j] = qmod(MOD + f1[j] - f1[pos]);
                f3[j] = qmod(MOD + f3[j] - f3[pos]);
            }
            last[j] = i;
        }
    }

    for (i = 1; i <= dn; i++) {
        if (d[i] > B) {
            f1[i] = qmod(f1[i] + qmod(MOD + p1sum[p[last[i]]] - p1sum[B]));
            f3[i] = qmod(f3[i] + qmod(MOD + p3sum[p[last[i]]] - p3sum[B]));
        }
        else {
            f1[i] = 1;
            f3[i] = 0;
        }
        f1[i] = qmod(MOD + f1[i] - 1);
        F2[i] = (1ll * f1[i] * POWK[3] + f3[i]) % MOD;
    }

    // #ifndef ONLINE_JUDGE
    //     printf("# F1 = %d\n", F1);
    //     for (i = 1; i <= dn; i++)
    //         printf("# d = %lld, f1 = %d, f3 = %d, F2 = %d\n", d[i], f1[i], f3[i], F2[i]);
    // #endif
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("3069.in", "r", stdin);
    freopen("3069.out", "w", stdout);
#endif

    int i, pos;
    int64 ans, x;

    scanf("%lld %lld %d", &n, &m, &MOD);

    if (n <= 3) {
        printf("%lld\n", 1ll * qpow(4, m) * n % MOD);
        return 0;
    }
    B = sqrt(n);
    linear_sleve();
    zz_sleve();

    ans = F1;
    for (i = 1; i <= B; i++) {
        x = n / i;
        pos = (x <= B) ? x : (dn - n / x + 1);
        ans = (ans + 1ll * g[i] * F2[pos]) % MOD;
    }

    printf("%lld\n", 1ll * POWK[4] * ans % MOD);
    return 0;
}
