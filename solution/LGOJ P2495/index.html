LGOJ #P2495. 「SDOI2011」消耗战

内存限制：512 MiB 时间限制：2000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

在一场战争中，战场由 $n$ 个岛屿和 $n-1$ 个桥梁组成，保证每两个岛屿间有且仅有一条路径可达。现在，我军已经侦查到敌军的总部在编号为 $1$ 的岛屿，而且他们已经没有足够多的能源维系战斗，我军胜利在望。已知在其他 $k$ 个岛屿上有丰富能源，为了防止敌军获取能源，我军的任务是炸毁一些桥梁，使得敌军不能到达任何能源丰富的岛屿。由于不同桥梁的材质和结构不同，所以炸毁不同的桥梁有不同的代价，我军希望在满足目标的同时使得总代价最小。

侦查部门还发现，敌军有一台神秘机器。即使我军切断所有能源之后，他们也可以用那台机器。机器产生的效果不仅仅会修复所有我军炸毁的桥梁，而且会重新随机资源分布（但可以保证的是，资源不会分布到 $1$ 号岛屿上）。不过侦查部门还发现了这台机器只能够使用 $m$ 次，所以我们只需要把每次任务完成即可。

输入格式

第一行一个整数 $n$，代表岛屿数量。

接下来 $n-1$ 行，每行三个整数 $u,v,w$，代表 $u$ 号岛屿和 $v$ 号岛屿由一条代价为 $c$ 的桥梁直接相连，保证 $1 \leq u,v \leq n$ 且 $1 \leq c \leq 100000$。

第 $n+1$ 行，一个整数 $m$，代表敌方机器能使用的次数。

接下来 $m$ 行，每行一个整数 $k_i$，代表第 $i$ 次后，有 $k_i$ 个岛屿资源丰富，接下来 $k$ 个整数 $h_1,h_2,\ldots,h_k$，表示资源丰富岛屿的编号。

输出格式

输出有 $m$ 行，分别代表每次任务的最小代价。

样例

输入样例

10
1 5 13
1 9 6
2 1 19
2 4 8
2 3 91
5 6 8
7 5 4
7 8 31
10 7 9
3
2 10 6
4 5 7 8 3
3 9 4 6

输出样例

12
32
22

数据范围与提示

对于 $10\%$ 的数据，$2 \leq n \leq 10,1 \leq m \leq 5,1 \leq k_i \leq n-1$

对于 $20\%$ 的数据，$2 \leq n \leq 100,1 \leq m \leq 100,1 \leq k_i \leq min(10,n-1)$

对于 $40\%$ 的数据，$2 \leq n \leq 1000,m \geq 1,\sum k_i \leq 500000,1 \leq k_i \leq \min(15,n-1)$

对于 $100\%$ 的数据，$2 \leq n \leq 250000,m \geq 1,\sum k_i \leq 500000,1 \leq k_i \leq n-1$

题解

虚树模板。

直接树形 DP 即可。

注意虚树连边时应当连 $u \rightarrow v$ 路径上的最小边权，而不是边权之和。

时间复杂度：$O((n + \sum k_i) \log n)$。

代码

#include <algorithm>
#include <cstdio>
#include <cstring>

typedef long long int64;

const int INF = 0x3f3f3f3f;

#ifdef ONLINE_JUDGE
const int MAXN = 250000 + 10;
const int MAXLOG = 19;
#else
const int MAXN = 30;
const int MAXLOG = 4;
#endif

int n, m;
int rt;

int pre[MAXN << 1], to[MAXN << 1], wdis[MAXN << 1], head[MAXN], wcnt;
int fa[MAXN][MAXLOG + 1], mindis[MAXN][MAXLOG + 1]; // 倍增用树上父亲 / 最小边权
int id[MAXN], dep[MAXN], idcnt;

int stk[MAXN], stop; // 建虚树用的栈
int qp[MAXN], qn;    // 储存询问点
int pp[MAXN], pn;    // 储存关键点（询问点及其 lca）
int64 f[MAXN];       // DP
bool ispp[MAXN];     // 是否为关键点

bool cmpbyid(int x, int y) { return id[x] < id[y]; }

inline void addedge(int u, int v, int w) {
    wcnt++;
    pre[wcnt] = head[u];
    head[u] = wcnt;
    to[wcnt] = v;
    wdis[wcnt] = w;
}
inline void add2edge(int u, int v, int w) {
    addedge(u, v, w);
    addedge(v, u, w);
}

#define v to[i]
void dfs1(int u) {
    int i;

    for (i = 1; fa[u][i - 1]; i++) {
        fa[u][i] = fa[fa[u][i - 1]][i - 1];
        mindis[u][i] = std::min(mindis[fa[u][i - 1]][i - 1], mindis[u][i - 1]);
    }

    idcnt++;
    id[u] = idcnt;
    for (i = head[u]; i; i = pre[i]) {
        if (!id[v]) {
            fa[v][0] = u;
            mindis[v][0] = wdis[i];
            dep[v] = dep[u] + 1;
            dfs1(v);
        }
    }
}

// f[u]: u 的子树（不含自己）全部切断的最小代价
void dfs2(int u) {
    int i;

    if (ispp[u]) {
        f[u] = INF;
        return;
    }
    for (i = head[u]; i; i = pre[i]) {
        dfs2(v);
        f[u] += std::min(1ll * wdis[i], f[v]);
    }
}
#undef v

inline int lca(int x, int y) {
    int i;

    if (dep[x] < dep[y])
        std::swap(x, y);
    for (i = MAXLOG; ~i; i--)
        if (dep[fa[x][i]] >= dep[y])
            x = fa[x][i];
    if (x == y)
        return x;
    for (i = MAXLOG; ~i; i--) {
        if (fa[x][i] ^ fa[y][i]) {
            x = fa[x][i];
            y = fa[y][i];
        }
    }
    return fa[x][0];
}
inline int querymin(int x, int f) {
    int i, res = INF;

    for (i = MAXLOG; ~i; i--) {
        if (dep[fa[x][i]] >= dep[f]) {
            res = std::min(res, mindis[x][i]);
            x = fa[x][i];
        }
    }
    return res;
}

#define adde(f, x) addedge(f, x, querymin(x, f))
void build() { // 建立虚树；进栈不连边，只在出栈时连边
    int i, x, z;

    std::sort(qp + 1, qp + 1 + qn, cmpbyid);
    stop = 1;
    pn = 1;
    stk[1] = 1;
    pp[1] = 1;
    for (i = 1; i <= qn; i++) {
        x = qp[i];
        pn++;
        pp[pn] = x;
        ispp[x] = true;

        while (stop > 1 and lca(stk[stop - 1], qp[i]) ^ stk[stop - 1]) {
            adde(stk[stop - 1], stk[stop]);
            stop--;
        }
        z = lca(x, stk[stop]);
        if (z ^ stk[stop]) {
            pn++;
            pp[pn] = z;
            adde(z, stk[stop]);
            stop--;
            if (stk[stop] ^ z) {
                stop++;
                stk[stop] = z;
            }
        }
        stop++;
        stk[stop] = x;
    }
    while (stop > 1) {
        adde(stk[stop - 1], stk[stop]);
        stop--;
    }
}
#undef adde

void clear() { // 清空数组
    int i;

    wcnt = 0;
    for (i = 1; i <= pn; i++) {
        ispp[pp[i]] = false;
        head[pp[i]] = f[pp[i]] = 0;
    }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("P2495.in", "r", stdin);
    freopen("P2495.out", "w", stdout);
#endif

    int i, u, v, w;

    scanf("%d", &n);
    for (i = 1; i < n; i++) {
        scanf("%d %d %d", &u, &v, &w);
        add2edge(u, v, w);
    }

    dep[1] = 1;
    dfs1(1);

    memset(head, 0, sizeof(head));
    wcnt = 0;

    scanf("%d", &m);
    while (m) {
        scanf("%d", &qn);
        for (i = 1; i <= qn; i++)
            scanf("%d", &qp[i]);

        build();
        dfs2(1);
        printf("%lld\n", f[1]);
        clear();
        m--;
    }
    return 0;
}
