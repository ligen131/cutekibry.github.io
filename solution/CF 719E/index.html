CF #719E. Sasha and Array

内存限制：256 MiB 时间限制：5000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

Sasha 有一个序列 $a_1, a_2, \ldots, a_n$。你需要进行 $m$ 个操作，操作有两种类型：

  1. $\mathtt{1\ l\ r\ x}$：将从 $l$ 到 $r$ 区间内的所有整数增加 $x$；
  2. $\mathtt{2\ l\ r}$：计算 $\sum_{i=l}^r f(a_i)$，$f(x)$ 表示第 $x$ 个斐波那契数。因为这个整数可能很大，你只需要计算它对 $10^9+7$ 取模的值。

本题中我们这样定义斐波那契数列：$f(1) = 1$，$f(2) = 1$，对于所有 $x > 2$ 有 $f(x) = f(x - 1) + f(x - 2)$。

Sasha 是一个才能出众的男孩，所以他只花了五秒就完成了所有操作。你能不能写一个能像 Sasha 那样棒的程序呢？

输入格式

第一行，两个整数 $n$ 和 $m$（$1 \leq n \leq 100000, 1 \leq m \leq 100000$），分别表示序列元素的数量以及操作个数。

下一行，$n$ 个整数 $a_1, a_2, \ldots, a_n$（$1 \leq a_i \leq 10^9$）。

接下来 $m$ 行，描述操作。每行有三到四个整数 $tp_i, l_i, r_i$ 和 $x_i$（$1 \leq tp_i \leq 2, 1 \leq l_i \leq r_i \leq n, 1 \leq x_i \leq 10^9$）。这里 $tp_i=1$ 代表第一种操作，$tp_i=2$ 代表第二种操作。

保证输入至少有一次第二种操作。

输出格式

对于每个询问，输出答案对 $10^9 + 7$ 取模后的值。

样例

输入样例

5 4
1 1 2 1 1
2 1 5
1 2 4 2
2 2 4
2 1 5

输出样例

5
7
9

样例解释

一开始，序列 $a$ 为 $1, 1, 2, 1, 1$。

第一个询问的答案为 $f(1) + f(1) + f(2) + f(1) + f(1) = 1 + 1 + 1 + 1 + 1 = 5$。

操作 $\mathtt{1\ 2\ 4\ 2}$ 后序列 $a$ 变为 $1, 3, 4, 3, 1$。

第二个询问的答案为 $f(3) + f(4) + f(3) = 2 + 3 + 2 = 7$。

第三个询问的答案为 $f(1) + f(3) + f(4) + f(3) + f(1) = 1 + 2 + 3 + 2 + 1 = 9$。

数据范围与提示

题目翻译：@Tsukimaru Oshawott

题解

用线段树维护矩阵乘法。

每个区间维护一个向量

$$ \begin{bmatrix} \sum f(a_i) \\ \sum f(a_i - 1) \\ \end{bmatrix} $$

修改时区间乘上

$$ \begin{bmatrix} 1 & 1 \\ 1 & 0 \\ \end{bmatrix} ^x $$

即可。

代码

#include <cstdio>

#define lp (p << 1)
#define rp (p << 1 | 1)
#define mid ((l + r) >> 1)
#define lson lp, l, mid
#define rson rp, mid + 1, r

typedef long long int64;
typedef int Matrix[3][3];

#ifdef ONLINE_JUDGE
const int MAXN = 100000 + 10;
#else
const int MAXN = 100;
#endif

const int MOD = 1000000007;

Matrix seg[MAXN << 2], mark[MAXN << 2];
int n, m;

inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
inline void mul(Matrix a, Matrix b) {
    static Matrix res;

    res[1][1] = ((int64)a[1][1] * b[1][1] + (int64)a[1][2] * b[2][1]) % MOD;
    res[1][2] = ((int64)a[1][1] * b[1][2] + (int64)a[1][2] * b[2][2]) % MOD;
    res[2][1] = ((int64)a[2][1] * b[1][1] + (int64)a[2][2] * b[2][1]) % MOD;
    res[2][2] = ((int64)a[2][1] * b[1][2] + (int64)a[2][2] * b[2][2]) % MOD;
    a[1][1] = res[1][1];
    a[1][2] = res[1][2];
    a[2][1] = res[2][1];
    a[2][2] = res[2][2];
}
inline void add(Matrix a, Matrix b) {
    a[1][1] = qmod(a[1][1] + b[1][1]);
    a[1][2] = qmod(a[1][2] + b[1][2]);
    a[2][1] = qmod(a[2][1] + b[2][1]);
    a[2][2] = qmod(a[2][2] + b[2][2]);
}
inline void fibpow(Matrix res, int p) {
    static Matrix b;

    res[1][1] = res[2][2] = 1;
    res[1][2] = res[2][1] = 0;
    b[1][1] = b[1][2] = b[2][1] = 1;
    b[2][2] = 0;
    while (p) {
        if (p & 1)
            mul(res, b);
        mul(b, b);
        p >>= 1;
    }
}

inline void addnode(int p, Matrix x) {
    mul(seg[p], x);
    mul(mark[p], x);
}
inline void down(int p) {
    if (mark[p][1][1] ^ 1 or mark[p][1][2] or mark[p][2][1] or mark[p][2][2] ^ 1) {
        addnode(lp, mark[p]);
        addnode(rp, mark[p]);
        mark[p][1][1] = mark[p][2][2] = 1;
        mark[p][1][2] = mark[p][2][1] = 0;
    }
}
inline void up(int p) {
    seg[p][1][1] = qmod(seg[lp][1][1] + seg[rp][1][1]);
    seg[p][1][2] = qmod(seg[lp][1][2] + seg[rp][1][2]);
    seg[p][2][1] = qmod(seg[lp][2][1] + seg[rp][2][1]);
    seg[p][2][2] = qmod(seg[lp][2][2] + seg[rp][2][2]);
}
void build(int p, int l, int r) {
    int x;

    mark[p][1][1] = mark[p][2][2] = 1;
    if (l ^ r) {
        build(lson);
        build(rson);
        up(p);
    }
    else {
        scanf("%d", &x);
        fibpow(seg[p], x - 1);
        seg[p][1][2] = seg[p][2][1];
        seg[p][2][1] = seg[p][2][2] = 0;
    }
}
void update(int p, int l, int r, int a, int b, Matrix x) {
    if (a <= l and r <= b) {
        addnode(p, x);
        return;
    }
    down(p);
    if (a <= mid)
        update(lson, a, b, x);
    if (b > mid)
        update(rson, a, b, x);
    up(p);
}
int query(int p, int l, int r, int a, int b) {
    if (a <= l and r <= b)
        return seg[p][1][1];
    down(p);
    if (a <= mid and b > mid)
        return qmod(query(lson, a, b) + query(rson, a, b));
    else if (a <= mid)
        return query(lson, a, b);
    else
        return query(rson, a, b);
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("e.in", "r", stdin);
    freopen("e.out", "w", stdout);
#endif

    static Matrix mat;
    int i, opt, l, r, x;

    scanf("%d %d", &n, &m);
    build(1, 1, n);
    for (i = 1; i <= m; i++) {
        scanf("%d %d %d", &opt, &l, &r);
        if (opt == 1) {
            scanf("%d", &x);
            fibpow(mat, x);
            update(1, 1, n, l, r, mat);
        }
        else
            printf("%d\n", query(1, 1, n, l, r));
    }
    return 0;
}
