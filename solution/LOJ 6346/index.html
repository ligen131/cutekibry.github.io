LOJ #6346. 线段树：关于时间

内存限制：256 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

有 $n$ 个整数 $a_1, a_2, \ldots, a_n$ 组成一个序列。有一个存储三元组的列表，开始时该列表为空。

有 $m$ 个操作，这些操作分为两种：

$\texttt{1 L R x}$ 将 $(L, R, x)$ 加入列表中。

$\texttt{2 L R}$ 求 $a_L + a_{L+1} + \ldots + a_R$ 。

每执行完一个操作，就读取一遍列表，对于其中的每一组 $(L, R, x)$，$a_L + a_{L+1} + \ldots + a_R$ 都加上 $x$（这不算做操作）。

输入格式

第一行一个整数 $n$，表示序列长度。

第二行 $n$ 个整数。

第三行一个整数 $m$，表示操作数。

然后 $m$ 行，先输入一个数 $D$，$D$ 为 $1$ 或 $2$ 。

若 $D$ 为 $1$，读入 $3$ 个整数 $L, R, x$。

若 $D$ 为 $2$，读入 $2$ 个整数 $L, R$。

输出格式

对于每个操作 $\texttt{2 L R}$，输出一行，一个整数，$a_L + a_{L+1} + \ldots + a_R$ 。

样例

样例输入

3
1 2 3
4
1 1 3 1
2 1 1
1 2 3 2
2 2 3

样例输出

2
15

样例解释

列表 输出 $a_i$
开始 $\texttt{1 2 3}$
$\texttt{1 1 3 1}$ $\texttt{1 3 1<}$ $\texttt{1 2 3}$
读取列表 $\texttt{1 3 1}$ $\texttt{2 3 4}$
$\texttt{2 1 1}$ $\texttt{1 3 1}$ $\texttt{2}$ $\texttt{2 3 4}$
读取列表 $\texttt{1 3 1}$ $\texttt{3 4 5}$
$\texttt{1 2 3 2}$ $\texttt{1 3 1}\\\texttt{2 3 2<}$ $\texttt{3 4 5}$
读取列表 $\texttt{1 3 1}\\\texttt{2 3 2}$ $\texttt{4 7 8}$
$\texttt{2 2 3}$ $\texttt{1 3 1}\\\texttt{2 3 2}$ $\texttt{15}$ $\texttt{4 7 8}$
读取列表 $\texttt{1 3 1}\\\texttt{2 3 2}$ $\texttt{5 10 11}$

数据范围与提示

对于 $60\%$ 的数据，暴力可过。

对于 $100\%$ 的数据，$1 \le n, m \le 10^5, 1 \le L \le R \le n, 1 \le a_i \le 10^4, -10^4 \le x \le 10^4$ 。

题解

很容易想到，我们只要在节点打上标记 $mark[p]$ ，表示该区间每秒增加 $mark[p]$ 的值。

每个操作开始时间不同？用 $cmark[p]$ 记录差分。

void add(int p, int l, int r, int a, int b, int64 c) {
    mark[p] += (std::min(r, b) - std::max(l, a) + 1) * c;
    cmark[p] -= (std::min(r, b) - std::max(l, a) + 1) * c * tcnt;
}

下面的代码使用了标记永久化。不使用也是可以的。

代码

#include <algorithm>
#include <cstdio>

#define lp (p << 1)
#define rp (p << 1 | 1)
#define mid ((l + r) >> 1)
#define lson lp, l, mid
#define rson rp, mid + 1, r
#define len (std::min(r, b) - std::max(l, a) + 1)

typedef long long int64;

const int MAXN = 100000 + 10;

int a[MAXN];
int64 mark[MAXN << 2], cmark[MAXN << 2], allmark[MAXN << 2],
    allcmark[MAXN << 2];
int tcnt;

void build(int p, int l, int r) {
    if (l ^ r) {
        build(lson);
        build(rson);
        cmark[p] = cmark[lp] + cmark[rp];
    }
    else
        cmark[p] = a[l];
}
void update(int p, int l, int r, int a, int b, int64 c) {
    mark[p] += len * c;
    cmark[p] -= len * c * tcnt;
    if (a <= l and r <= b) {
        allmark[p] += c;
        allcmark[p] -= c * tcnt;
        return;
    }
    if (a <= mid)
        update(lson, a, b, c);
    if (b> mid)
        update(rson, a, b, c);
}
int64 query(int p, int l, int r, int a, int b) {
    int64 res = len * (allmark[p] * tcnt + allcmark[p]);
    if (a <= l and r <= b)
        return mark[p] * tcnt + cmark[p];
    if (a <= mid)
        res += query(lson, a, b);
    if (b> mid)
        res += query(rson, a, b);
    return res;
}

int main() {
    register int i, n, m, opt, l, r, x;

    scanf("%d", &n);
    for (i = 1; i <= n; i++)
        scanf("%d", &a[i]);
    build(1, 1, n);

    scanf("%d", &m);
    for (i = 1; i <= m; i++) {
        tcnt = i;
        scanf("%d %d %d", &opt, &l, &r);
        if (opt == 1) {
            scanf("%d", &x);
            update(1, 1, n, l, r, x);
        }
        else
            printf("%lld\n", query(1, 1, n, l, r));
    }

    return 0;
}
