LOJ #2466. 「POI2014」卡片 Card

内存限制：256 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

桌面上有依次排列的 $n$ 张卡片，每张卡片有正反两面。共 $m$ 次操作，每次操作会交换两张卡片，询问交换后是否有可能通过改变卡片的正反来使卡片上的数不降。

输入格式

第一行一个整数 $n$ ，表示卡片的数量。

接下来 $n$ 行，每行两个整数 $x_i,y_i$ ，表示卡片的正反两面。

接下来 $m$ 行，第 $j$ 行有两个整数 $a_j,b_j$，表示第 $j$ 次操作交换第 $a$ 张和 $b$ 张卡片。

输出格式

输出 $m$ 行，第 $j$ 行表示第 $j$ 次操作后是否能使卡片上的数不降，输出一行字符串 "TAK" 或 "NIE"，分别表示可能、不可能。

样例

样例输入

4
2 5
3 4
6 3
2 7
2
3 4
1 3

样例输出

NIE
TAK

数据范围与提示

对于 $30\%$ 的数据，保证每张卡两面的数字相同。

对于 $38\%$ 的数据（可能不同），$n \le 2000$ 。

对于 $100\%$ 的数据， $2 \le n \le 200000 , 0 \le x_i,y_i \le 10^7 , 1 \le m \le 1000000$ 。

对于下发的三组样例：

  • 1ocen：$n=6,m=6$，用于测试程序正确性。
  • 2ocen：$n=7,m=9$，保证卡的两面数字相同。
  • 3ocen：$n=200000,m=1000000$，所有偶数编号的操作反转其前一次操作。

题解

假设 $x_i \leq y_i$。

我们令 $f(i)$ 为我们选择的第 $i$ 张卡片的整数，那么有：

$$ f(i) = \begin{cases} 0, &i = 0 \\ x_i, &i > 0 \cap f(i-1) \leq x_i \\ y_i, &i > 0 \cap x_i < f(i-1) \leq y_i \\ \infty, &i > 0 \cap f(i-1) > y_i \\ \end{cases} $$

只要判断 $f(n)$ 是否等于 $\infty$ 即可。

使用线段树维护。

设线段树上节点 $p$ 代表区间 $[l, r]$，则我们要维护 $sf[p][0], sf[p][1]$ 分别表示当 $f(l)=x_i, f(l)=y_i$ 时 $f(r)$ 的值。

单点修改和总体询问。

时间复杂度：$O(m \log n)$。

代码

#include <algorithm>
#include <cstdio>

#define lp (p << 1)
#define rp (p << 1 | 1)
#define mid ((ll[p] + rr[p]) >> 1)

const int MAXN = 200000 + 10;
// const int MAXN = 1000 + 10;
const int INF = 0x3f3f3f3f;

int n, m;
int a[MAXN], b[MAXN];
int sf[MAXN << 2][2];
int ll[MAXN << 2], rr[MAXN << 2];
int pos[MAXN];

inline void up(int p) {
    if (sf[lp][0] <= a[mid + 1])
        sf[p][0] = sf[rp][0];
    else if (sf[lp][0] <= b[mid + 1])
        sf[p][0] = sf[rp][1];
    else
        sf[p][0] = INF;

    if (sf[lp][1] <= a[mid + 1])
        sf[p][1] = sf[rp][0];
    else if (sf[lp][1] <= b[mid + 1])
        sf[p][1] = sf[rp][1];
    else
        sf[p][1] = INF;
}
void build(int p, int l, int r) {
    ll[p] = l;
    rr[p] = r;
    if (l ^ r) {
        build(lp, l, mid);
        build(rp, mid + 1, r);
        up(p);
    }
    else {
        pos[l] = p;
        sf[p][0] = a[l];
        sf[p][1] = b[l];
    }
}

inline void update(int p, int x, int y) {
    a[p] = sf[pos[p]][0] = x;
    b[p] = sf[pos[p]][1] = y;
    for (p = pos[p] >> 1; p; p >>= 1)
        up(p);
}

int main() {
#ifndef ONLINE_JUDGE

    freopen("2466.out", "w", stdout);
#endif

    int i, x, y, a1, b1, a2, b2;

    scanf("%d", &n);
    for (i = 1; i <= n; i++) {
        scanf("%d %d", &a[i], &b[i]);
        if (a[i] > b[i])
            std::swap(a[i], b[i]);
    }

    build(1, 1, n);

    scanf("%d", &m);
    for (i = 1; i <= m; i++) {
        scanf("%d %d", &x, &y);
        a1 = a[x];
        b1 = b[x];
        a2 = a[y];
        b2 = b[y];
        update(x, a2, b2);
        update(y, a1, b1);
        printf((sf[1][0] ^ INF) ? "TAK\n" : "NIE\n");
    }
    return 0;
}
