LOJ #2032. 「SDOI2016」游戏

内存限制：256 MiB 时间限制：2000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

Alice 和 Bob 在玩一个游戏。

游戏在一棵有 $n$ 个点的树上进行。最初，每个点上都只有一个数字，那个数字是 $123456789123456789$。

有时，Alice 会选择一条从 $s$ 到 $t$ 的路径，在这条路径上的每一个点上都添加一个数字。对于路径上的一个点 $r$，若 $r$ 与 $s$ 的距离是 $\mathrm{dis}$，那么 Alice 在点 $r$ 上添加的数字是 $a\cdot \mathrm{dis}+b$。有时，Bob 会选择一条从 $s$ 到 $t$ 的路径。他需要先从这条路径上选择一个点，再从那个点上选择一个数字。

Bob 选择的数字越小越好，但大量的数字让 Bob 眼花缭乱。

Bob 需要你帮他找出他能够选择的最小的数字。

输入格式

第一行两个数字 $n,m$，表示树的点数和进行的操作数。

接下来 $n−1$ 行，每行三个数字 $u,v,w$，表示树上有一条连接 $u,v$ 的边，长度是 $w$。

接下来 $m$ 行。每行第一个数字是 $1$ 或 $2$。

若第一个数是 $1$，表示 Alice 进行操作，接下来四个数字 $s,t,a,b$。

若第一个数是 $2$，表示 Bob 进行操作，接下来两个数字 $s,t$。

输出格式

每当 Bob 进行操作，输出一行一个数，表示他能够选择的最小的数字。

样例

样例输入

3 5
1 2 10
2 3 20
2 1 3
1 2 3 5 6
2 2 3
1 2 3 -5 -6
2 2 3

样例输出

123456789123456789
6
-106

数据范围与提示

测试点 1 ~ 2：$ n \leq 10 $，$ m \leq 10 $，$ \left| a \right| \leq 10000 $；

测试点 3 ~ 4：$ n \leq 1000 $，$ m \leq 1000 $，$ \left| a \right| \leq 10000 $；

测试点 5：$ n \leq 100000 $，$ m \leq 100000 $，$ a = 0 $，树是一条链；

测试点 6 ~ 7：$ n \leq 100000 $，$ m \leq 100000 $，$ a = 0 $；

测试点 8：$ n \leq 100000 $，$ m \leq 100000 $，$ a = 1 $，树是一条链；

测试点 9 ~ 10：$ n \leq 100000 $，$ m \leq 100000 $，$ a = 1 $；

测试点 11 ~ 13：$ n \leq 100000 $，$ m \leq 100000 $，$ \left| a \right| \leq 10000 $，树是一条链；

测试点 14 ~ 20：$ n \leq 100000 $，$ m \leq 100000 $，$ \left| a \right| \leq 10000 $。

题解

各变量定义：

  • $dis_p$：节点 $p$ 到根结点的路径长度。
  • $dis_{u \rightarrow v}$：节点 $u$ 到节点 $v$ 的路径长度。
  • $dep_p$：节点 $p$ 的深度。

考虑把修改拆成 $s \rightarrow lca(s, t)$ 和 $t \rightarrow lca(s, t)$ 进行处理。

对于 $s \rightarrow lca(s, t)$，$f(p) = (dis_s - dis_p)a + b = -adis_p + (dis_sa + b)$；

对于 $t \rightarrow lca(s, t)$，$f(p) = (dis_{s \rightarrow t} - (dis_t - dis_p))a + b = adis_p + (dis_{s \rightarrow t}a + -dis_ta + b)$。

容易发现，这是一个一次函数。用树链剖分和李超线段树维护即可。

时间复杂度：$O(n \log^3 n)$。

代码

#include <algorithm>
#include <cstdio>

#define lp (p << 1)
#define rp (p << 1 | 1)
#define mid ((l + r) >> 1)
#define lson lp, l, mid
#define rson rp, mid + 1, r

typedef long long int64;

#ifdef ONLINE_JUDGE
const int MAXN = 100000 + 10;
#else
const int MAXN = 1000;
#endif

const int64 INF = 123456789123456789;

int n, m;

int64 segk[MAXN << 2], segb[MAXN << 2];
int64 minv[MAXN << 2];

int pre[MAXN << 1], to[MAXN << 1], wdis[MAXN << 1], head[MAXN], wcnt;
int size[MAXN], far[MAXN], son[MAXN], dep[MAXN];
int64 dis[MAXN];
int top[MAXN], id[MAXN], rid[MAXN], idcnt;

inline void up(int p, int l, int r) {
    minv[p] = (segk[p] > 0) ? (segk[p] * dis[rid[l]] + segb[p]) : (segk[p] * dis[rid[r]] + segb[p]);
    if (l ^ r)
        minv[p] = std::min(minv[p], std::min(minv[lp], minv[rp]));
}
void addnode(int p, int l, int r, int64 k, int64 b) {
    if (segk[p] * dis[rid[l]] + segb[p] > k * dis[rid[l]] + b) {
        std::swap(segk[p], k);
        std::swap(segb[p], b);
    }
    if (k == segk[p] or l == r) {
        up(p, l, r);
        return;
    }

    double x = double(segb[p] - b) / (k - segk[p]);
    if (x < dis[rid[l]] or x > dis[rid[r]]) {
        up(p, l, r);
        return;
    }
    else if (x <= dis[rid[mid]]) {
        addnode(lson, segk[p], segb[p]);
        segk[p] = k;
        segb[p] = b;
    }
    else
        addnode(rson, k, b);
    up(p, l, r);
}
void update(int p, int l, int r, int a, int b, int64 kk, int64 bb) {
    if (a <= l and r <= b) {
        addnode(p, l, r, kk, bb);
        return;
    }
    if (a <= mid)
        update(lson, a, b, kk, bb);
    if (b > mid)
        update(rson, a, b, kk, bb);
    up(p, l, r);
}
int64 query(int p, int l, int r, int a, int b) {
    if (a <= l and r <= b)
        return minv[p];
    int64 res = (segk[p] > 0) ? (segk[p] * dis[rid[std::max(l, a)]] + segb[p])
                              : (segk[p] * dis[rid[std::min(r, b)]] + segb[p]);
    if (a <= mid) {
        addnode(lson, segk[p], segb[p]);
        res = std::min(res, query(lson, a, b));
    }
    if (b > mid) {
        addnode(rson, segk[p], segb[p]);
        res = std::min(res, query(rson, a, b));
    }
    return res;
}

inline void addedge(int u, int v, int w) {
    wcnt++;
    pre[wcnt] = head[u];
    head[u] = wcnt;
    to[wcnt] = v;
    wdis[wcnt] = w;
}
inline void add2edge(int u, int v, int w) {
    addedge(u, v, w);
    addedge(v, u, w);
}
#define v to[i]
void dfs1(int u) {
    int i;

    size[u] = 1;
    for (i = head[u]; i; i = pre[i]) {
        if (!size[v]) {
            far[v] = u;
            dis[v] = dis[u] + wdis[i];
            dep[v] = dep[u] + 1;
            dfs1(v);
            size[u] += size[v];
            if (size[son[u]] < size[v])
                son[u] = v;
        }
    }
}
void dfs2(int u, int topf) {
    int i;

    idcnt++;
    id[u] = idcnt;
    rid[idcnt] = u;
    top[u] = topf;
    if (son[u]) {
        dfs2(son[u], topf);
        for (i = head[u]; i; i = pre[i])
            if (!id[v])
                dfs2(v, v);
    }
}
int lca(int x, int y) {
    while (top[x] ^ top[y]) {
        if (dep[top[x]] < dep[top[y]])
            std::swap(x, y);
        x = far[top[x]];
    }
    if (dep[x] > dep[y])
        std::swap(x, y);
    return x;
}
// id[top[x]] -> id[x]
// id[x]:   realval ldis * a + b
//          infact  -id[x] * a + b
//          fixed   -id[x] * a + b + (ldis + id[x]) * a

// id[top[y]] -> id[y]
// id[y]:   realval rdis * a + b
//          infact  id[y] * a + b
//          fixed   id[y] * a + b + (rdis - id[y]) * a

void update(int x, int y, int64 a, int64 b) {
    int z, p;
    int64 xydis;

    z = lca(x, y);
    xydis = dis[x] + dis[y] - 2 * dis[z];

    p = x;
    while (top[p] ^ top[z]) {
        update(1, 1, n, id[top[p]], id[p], -a, dis[x] * a + b);
        p = far[top[p]];
    }
    update(1, 1, n, id[z], id[p], -a, dis[x] * a + b);

    p = y;
    while (top[p] ^ top[z]) {
        update(1, 1, n, id[top[p]], id[p], a, (xydis - dis[y]) * a + b);
        p = far[top[p]];
    }
    update(1, 1, n, id[z], id[p], a, (xydis - dis[y]) * a + b);
}
int64 query(int x, int y) {
    int64 ans = INF;

    while (top[x] ^ top[y]) {
        if (dep[top[x]] < dep[top[y]])
            std::swap(x, y);
        ans = std::min(ans, query(1, 1, n, id[top[x]], id[x]));
        x = far[top[x]];
    }
    if (dep[x] > dep[y])
        std::swap(x, y);
    ans = std::min(ans, query(1, 1, n, id[x], id[y]));
    return ans;
}
#undef v

int main() {
#ifndef ONLINE_JUDGE
    freopen("2032.in", "r", stdin);
    freopen("2032.out", "w", stdout);
#endif

    int i, opt, u, v, x, y;

    scanf("%d %d", &n, &m);
    for (i = 1; i < n; i++) {
        scanf("%d %d %d", &u, &v, &x);
        add2edge(u, v, x);
    }

    for (i = 1; i <= n << 2; i++)
        minv[i] = segb[i] = INF;
    dfs1(1);
    dfs2(1, 1);

    for (i = 1; i <= m; i++) {
        scanf("%d %d %d", &opt, &u, &v);
        if (opt == 1) {
            scanf("%d %d", &x, &y);
            update(u, v, x, y);
        }
        else
            printf("%lld\n", query(u, v));
    }
}
