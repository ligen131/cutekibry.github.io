LOJ #556. 「Antileaf's Round」咱们去烧菜吧

内存限制：512 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

你有 $m$ 种物品，第 $i$ 种物品的大小为 $a_i$，数量为 $b_i$（$b_i=0$ 表示有无限个）。

你还有 $n$ 个背包，体积分别为 $1$ 到 $n$，现在你很想知道用这些物品填满某个背包的方案数。

为了满足你的好奇心，你决定把填满每个背包的方案数都算一遍。

因为你其实只是闲得无聊，所以你只想知道方案数对 $998244353$（$7\times 17\times 2^{23}+1$，一个质数）取模后的值。

输入格式

第一行两个非负整数，分别表示 $n,m$。

以下 $m$ 行，每行两个非负整数，分别表示 $a_i,b_i$。

输出格式

输出 $n$ 个非负整数表示答案。

样例

样例输入

5 3
1 0
1 1
3 2

样例输出

2
2
3
4
4

样例解释

拼出 $1$~$5$ 的方案分别如下：

$\{1_1\},\{1_2\}$$\{1_1,1_1\},\{1_1,1_2\}$$\{1_1,1_1,1_1\},\{1_1,1_1,1_2\},\{3\}$$\{1_1,1_1,1_1,1_1\},\{1_1,1_1,1_1,1_2\},\{1_1,3\},\{1_2,3\}$$\{1_1,1_1,1_1,1_1,1_1\},\{1_1,1_1,1_1,1_1,1_2\},\{1_1,1_1,3\},\{1_1,1_2,3\}$

数据范围与提示

$0< n,m\le 10^5, 0\le a_i\le 110000,0\le b_i\le 10^6$。

题解

显然可以设生成函数

$$ \begin{aligned} F(x) &= \prod_{i = 1}^m \sum_{j = 0}^{b_i} a_ix^j \\ &= \prod_{i = 1}^m \frac {1 - x^{(b_i + 1)a_i}}{1 - x^{a_i}} \\ \end{aligned} $$

我们可以把它表示为 $G(x) = \prod_{i = 1}^n (1 - x^i)^{c_i}$ 的形式。考虑如何求出 $G(x)$。

$$ \begin{aligned} G(x) &= \prod_{i = 1}^n (1 - x^i)^{c_i} \\ \ln G(x) &= \ln \prod_{i = 1}^n (1 - x^i)^{c_i} \\ \ln G(x) &= \sum_{i = 1}^n c_i \ln (1 - x^i) \\ \ln G(x) &= \sum_{i = 1}^n c_i \sum_{k = 1}^{\infty} \frac {x^{ik}}{k} \\ \end{aligned} $$

直接计算 $\ln G(x)$，然后对 $\ln G(x)$ 做一遍多项式 $\exp$ 即可。

时间复杂度：$O(n \log n + m)$。

代码

#include <algorithm>
#include <cstdio>
#include <cstring>

typedef long long int64;

const int MOD = 998244353;
const int G = 3;
const int INVG = 332748118;

const int MAXLOG = 20;
const int MAXN = 1 << MAXLOG;

typedef int Expr[MAXN];

int n, m, len;

int c[MAXN];

int rev[MAXN];
int INV[MAXN];
int GT[MAXLOG], INVGT[MAXLOG];

inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
inline int qpow(int64 b, int p, int mod) {
    int res = 1;
    while (p) {
        if (p & 1)
            res = res * b % mod;
        b = b * b % mod;
        p >>= 1;
    }
    return res;
}
inline void init() {
    int i;
    int64 x;

    for (i = 1; i < MAXLOG; i++) {
        GT[i] = qpow(G, (MOD - 1) / (1 << i), MOD);
        INVGT[i] = qpow(INVG, (MOD - 1) / (1 << i), MOD);
    }
    x = 1;
    for (i = 2; i < MAXN; i++)
        x = x * i % MOD;
    INV[MAXN - 1] = qpow(x, MOD - 2, MOD);
    for (i = MAXN - 2; i; i--)
        INV[i] = INV[i + 1] * (i + 1ll) % MOD;
    x = 1;
    for (i = 2; i < MAXN; x = x * i % MOD, i++)
        INV[i] = INV[i] * x % MOD;
}
inline void initrev(int len) {
    int i, k;

    k = 1;
    while (len >> k >> 1)
        k++;
    for (i = 1; i < len; i++)
        rev[i] = (rev[i >> 1] >> 1) | ((i & 1) << (k - 1));
}
void dft(Expr a, int len, bool inv = false) {
    int i, j, k, logk;
    int x, y;
    int64 g, gn;

    for (i = 1; i < len; i++)
        if (i < rev[i])
            std::swap(a[i], a[rev[i]]);
    for (k = 1, logk = 1; k < len; k <<= 1, logk++) {
        gn = inv ? INVGT[logk] : GT[logk];
        for (i = 0; i < len; i += k << 1) {
            g = 1;
            for (j = 0; j < k; j++) {
                x = a[i + j];
                y = g * a[i + j + k] % MOD;
                a[i + j] = qmod(x + y);
                a[i + j + k] = qmod(x - y + MOD);
                g = g * gn % MOD;
            }
        }
    }
    if (inv) {
        g = qpow(len, MOD - 2, MOD);
        for (i = 0; i < len; i++)
            a[i] = a[i] * g % MOD;
    }
}
#define idft(a, len) dft(a, len, true)

inline void det(Expr a, Expr b, int len) {
    int i;
    for (i = 1; i < len; i++)
        a[i - 1] = 1ll * b[i] * i % MOD;
    a[len - 1] = 0;
}
inline void int_(Expr a, Expr b, int len) {
    int i;
    for (i = len - 1; i; i--)
        a[i] = 1ll * b[i - 1] * INV[i] % MOD;
    a[0] = 0;
}
void inv(Expr res, Expr b, int len) {
    static Expr a, bb;
    int i, k;

    a[0] = qpow(b[0], MOD - 2, MOD);
    for (k = 4; k < len << 1; k <<= 1) {
        memcpy(bb, b, k << 1);

        initrev(k);
        dft(a, k);
        dft(bb, k);
        for (i = 0; i < k; i++)
            a[i] = 1ll * a[i] * qmod(MOD + 2 - 1ll * a[i] * bb[i] % MOD) % MOD;
        idft(a, k);
        memset(a + (k >> 1), 0, k << 1);
    }

    memcpy(res, a, k << 2);
    memset(a, 0, k << 2);
    memset(bb, 0, k << 2);
}
void ln(Expr res, Expr b, int len) {
    static Expr a, inva;
    int i;

    len <<= 1;

    memcpy(a, b, len << 1);
    det(a, a, len >> 1);
    inv(inva, b, len >> 1);

    initrev(len);
    dft(a, len);
    dft(inva, len);
    for (i = 0; i < len; i++)
        res[i] = 1ll * a[i] * inva[i] % MOD;
    idft(res, len);
    int_(res, res, len);

    memset(res + (len >> 1), 0, len << 1);
    memset(a, 0, len << 2);
    memset(inva, 0, len << 2);
}
void exp(Expr res, Expr b, int len) {
    static Expr a, lna;
    int i, k;

    a[0] = 1;
    for (k = 4; k < len << 2; k <<= 1) {
        ln(lna, a, k);
        memset(lna + (k >> 1), 0, k << 1);
        for (i = 0; i < (k >> 1); i++)
            lna[i] = qmod(MOD - lna[i] + b[i]);
        lna[0] = qmod(lna[0] + 1);

        initrev(k);
        dft(a, k);
        dft(lna, k);
        for (i = 0; i < k; i++)
            a[i] = 1ll * a[i] * lna[i] % MOD;
        idft(a, k);
        memset(a + (k >> 1), 0, k << 1);
    }
    memcpy(res, a, k << 2);
    memset(a, 0, k << 2);
    memset(lna, 0, k << 2);
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("556.in", "r", stdin);
    freopen("556.out", "w", stdout);
#endif

    int i, j, x, y;
    static Expr f, g;

    init();

    scanf("%d %d", &n, &m);
    len = 1;
    while (len <= n)
        len <<= 1;
    for (i = 1; i <= m; i++) {
        scanf("%d %d", &x, &y);
        if (!y)
            y = n;
        if (x) {
            if (x <= n)
                c[x]++;
            if (1ll * (y + 1) * x <= n) {
                c[(y + 1) * x]--;
            }
        }
    }
    for (i = 1; i <= n; i++) {
        c[i] = qmod(MOD + c[i]);
        for (j = 1; i * j <= n; j++)
            f[i * j] = (f[i * j] + 1ll * c[i] * INV[j]) % MOD;
    }
    exp(g, f, len);
    for (i = 1; i <= n; i++)
        printf("%d\n", g[i]);
    return 0;
}
