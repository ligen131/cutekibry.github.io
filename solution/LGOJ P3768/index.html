LGOJ #P3768. 简单的数学题

题目类型：传统 评测方式：文本比较
上传者：未知

题目描述

由于出题人懒得写背景了，题目还是简单一点好。

输入一个整数 $n$ 和一个整数 $p$ ，你需要求出 $(\sum_{i=1}^n\sum_{j=1}^n ij\gcd(i,j)) \bmod p$，其中 $\gcd(a,b)$ 表示 $a$ 与 $b$ 的最大公约数。

输入格式

一行两个整数 $p,n$。

输出格式

一行一个整数 $(\sum_{i=1}^n\sum_{j=1}^n ij\gcd(i,j)) \bmod p$。

样例

样例输入

998244353 2000

样例输出

883968974

数据范围与提示

对于 $20\%$ 的数据，$n \leq 1000$。

对于 $30\%$ 的数据，$n \leq 5000$。

对于 $60\%$ 的数据，$n \leq 10^6$，时限 1s。

对于另外 $20\%$ 的数据，$n \leq 10^9$，时限 3s。

对于最后 $20\%$ 的数据，$n \leq 10^10$，时限 6s。

对于 $100\%$ 的数据，$5 \times 10^8 \leq p \leq 1.1 \times 10^9$ 且 $p$ 为素数。

题解

同样，先莫比乌斯变换一下。

$$ f(x)=\sum_{i=1}^n \sum_{j=1}^n ij[\gcd(i, j)=x] \\ F(x)=\sum_{i=1}^n \sum_{j=1}^n ij[x|\gcd(i, j)] \\ $$

算一下 $F(x)$ 的公式：

$$ \begin{aligned} F(x) &=\sum_{i=1}^n \sum_{j=1}^n ij[x|\gcd(i, j)] \\ &=\sum_{i=1}^{\lfloor \frac nx \rfloor} \sum_{j=1}^{\lfloor \frac nx \rfloor} ixjx \\ &=x^2 \sum_{i=1}^{\lfloor \frac nx \rfloor} i \sum_{j=1}^{\lfloor \frac nx \rfloor} j \\ &=x^2 (\sum_{i=1}^{\lfloor \frac nx \rfloor} i)^2 \\ \end{aligned} $$

答案就是

$$ \begin{aligned} ans &=\sum_{i=1}^n i f(i) \\ &=\sum_{i=1}^n i \sum_{i|d} \mu(\frac di) F(d) \\ &=\sum_{i=1}^n i \sum_{j=1}^{\lfloor \frac ni \rfloor} \mu(j) F(ij) \\ \end{aligned} $$

同样地，按照一般套路，设 $T=ij$。

$$ \begin{aligned} ans &=\sum_{i=1}^n i \sum_{j=1}^{\lfloor \frac ni \rfloor} \mu(j) F(ij) \\ &=\sum_{T=1}^n \sum_{i|T} i \mu(\frac Ti) F(T) \\ &=\sum_{T=1}^n F(T) \sum_{i|T} i \mu(\frac Ti) \\ &=\sum_{T=1}^n F(T) \varphi(T)\\ &=\sum_{T=1}^n (\sum_{i=1}^{\lfloor \frac nx \rfloor} i)^2 T^2 \varphi(T) \\ \end{aligned} $$

数论分块一下，那么我们只需要解决 $f=id^2 \cdot \varphi$ 的前缀和即可。

若 $f$ 是完全积性函数，$g,h$ 是数论函数，则 $(f\cdot g) \: * \: (f\cdot h)=f\cdot(g \: * \: h)$。

构造 $g=id^2 \cdot I$。

$$ f \: * \: g = h \\ (id^2 \cdot \varphi) \: * \: (id^2 \cdot I) = h \\ id^2 \cdot (\varphi \: * \: I) = h \\ id^3 = h \\ $$

令 $f=id^2 \cdot \varphi, g=id^2 \cdot I, h=id^3$ 即可。

时间复杂度：$O(n^{\frac 23})$。

代码

#include <cstdio>

typedef long long int64;

#ifdef ONLINE_JUDGE
const int B = 30000000 + 10;
#else
const int B = 1;
// const int B = 1001;
#endif
const int HMOD = 19260817;
const int MAXSIZE = 30000000 + 10;

int MOD, inv2, inv6;
int64 n;

int prefsum[B];

int pre[MAXSIZE], head[HMOD], cnt;
int64 key[MAXSIZE];
int val[MAXSIZE];

inline int64 min(int64 x, int64 y) { return x < y ? x : y; }

inline int gethash(int64 x) { return x % HMOD; }
inline int find(int64 k) {
    int h = gethash(k), p;
    p = head[h];
    while (p and key[p] ^ k)
        p = pre[p];
    return p;
}
inline void insert(int64 k, int v) {
    int h = gethash(k);
    cnt++;
    pre[cnt] = head[h];
    head[h] = cnt;
    key[cnt] = k;
    val[cnt] = v;
}

inline int qpow(int b, int p, int mod) {
    int res = 1;

    while (p) {
        if (p & 1)
            res = (int64)res * b % mod;
        b = (int64)b * b % mod;
        p >>= 1;
    }
    return res;
}
void init() {
    static int p[B];
    static bool notp[B];
    int pn;
    int i, j;

    pn = 0;
    prefsum[1] = 1;
    for (i = 2; i < min(B, n + 1); i++) {
        if (!notp[i]) {
            pn++;
            p[pn] = i;
            prefsum[i] = i - 1;
        }
        for (j = 1; j <= pn and i * p[j] < min(B, n + 1); j++) {
            notp[i * p[j]] = true;
            if (i % p[j])
                prefsum[i * p[j]] = (int64)prefsum[i] * prefsum[p[j]] % MOD;
            else {
                prefsum[i * p[j]] = (int64)prefsum[i] * p[j] % MOD;
                break;
            }
        }
    }
    for (i = 1; i < min(B, n + 1); i++)
        prefsum[i] = (prefsum[i - 1] + (int64)prefsum[i] * i % MOD * i) % MOD;
}
inline int gsum(int64 n) {
    n %= MOD;
    return (int64)n * (n + 1) % MOD * (2ll * n + 1) % MOD * inv6 % MOD;
}
inline int hsum(int64 n) {
    n %= MOD;
    int res = (int64)n * (n + 1) % MOD * inv2 % MOD;
    return (int64)res * res % MOD;
}
int fsum(int64 n) {
    int res;
    int64 i, j;
    int presum, nowsum;

    if (n < min(B, n + 1))
        return prefsum[n];
    else if (find(n))
        return val[find(n)];
    res = hsum(n);
    presum = MOD - 1;
    for (i = 2; i <= n; i = j + 1) {
        j = n / (n / i);
        nowsum = gsum(j);
        res = (res - ((int64)nowsum + presum) * fsum(n / i) % MOD + MOD) % MOD;
        presum = MOD - nowsum;
    }
    insert(n, res);
    return res;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("P3768.in", "r", stdin);
    freopen("P3768.out", "w", stdout);
#endif

    int64 i, j;
    int presum, nowsum, msum, res = 0;

    scanf("%d %lld", &MOD, &n);
    inv2 = qpow(2, MOD - 2, MOD);
    inv6 = qpow(6, MOD - 2, MOD);
    init();

    presum = 0;
    for (i = 1; i <= n; i = j + 1) {
        j = n / (n / i);
        nowsum = fsum(j);
        msum = hsum(n / i);
        res = (res + msum * ((int64)nowsum + presum)) % MOD;
        presum = MOD - nowsum;
        // printf("%d %d\n", j, nowsum);
    }
    printf("%d\n", res);
    return 0;
}
