LOJ #6018. Shlw loves matrix II

内存限制：256 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

给定矩阵 $M$，请计算 $M^n$，并将其中每一个元素对 $1000000007$ 取模输出。

输入格式

第一行包含两个整数 $n,k$，其中 $n$ 使用二进制表示，可能含有前导零

余下 $k$ 行描述了一个 $k \times k$ 的矩阵 $M$。

输出格式

$k$ 行输出答案矩阵

样例

输入样例

010 3
5 9 5
5 4 0
8 8 8

输出样例

110 121 65
45 61 25
144 168 104

数据范围与提示

$k\leq 50,n\leq 2^{10000}$

题解

主页面：「专题」线性递推

模板题。

时间复杂度：$O(k^4 + k^2 (\log n + \log P))$，其中 $P = 1000000007$ 为模数。

代码

#include <algorithm>
#include <cstdio>
#include <cstring>

const int MAXK = 100 + 10;
const int MAXN = 10000 + 10;
const int MOD = 1000000007;

typedef int Matrix[MAXK][MAXK];
typedef int Expr[MAXK];
typedef long long int64;

int m;
char s[MAXN];
Matrix mat;
Expr b, res, modexpr;

inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
inline int qpow(int64 b, int p, int mod = MOD) {
    int res = 1;

    while (p) {
        if (p & 1)
            res = res * b % mod;
        b = b * b % mod;
        p >>= 1;
    }
    return res;
}

void solve(Matrix a) {
    int i, j, k;
    int64 inv, p;

    for (i = 1; i <= m + 1; i++) {
        inv = qpow(a[i][i], MOD - 2);
        for (j = i + 1; j <= m + 1; j++) {
            p = qmod(MOD - inv * a[j][i] % MOD);
            for (k = i; k <= m + 2; k++)
                a[j][k] = (a[j][k] + p * a[i][k]) % MOD;
        }
    }

    for (i = m + 1; i; i--) {
        a[i][m + 2] = 1ll * a[i][m + 2] * qpow(a[i][i], MOD - 2) % MOD;
        for (j = i - 1; j; j--)
            a[j][m + 2] = qmod(MOD + a[j][m + 2] - 1ll * a[j][i] * a[i][m + 2] % MOD);
    }
}

int det(Matrix a) {
    int i, j, k, res = 1;
    int64 inv, p;
    Matrix temp;

    memcpy(temp, a, sizeof(Matrix));
    for (i = 1; i <= m; i++) {
        if (!temp[i][i]) {
            j = i + 1;
            while (j <= m and !temp[j][i])
                j++;
            if (j > m)
                return 0;
            for (k = i; k <= m; k++)
                std::swap(temp[i][k], temp[j][k]);
            res = qmod(MOD - res);
        }

        res = 1ll * res * temp[i][i] % MOD;
        inv = qpow(temp[i][i], MOD - 2);
        for (j = i + 1; j <= m; j++) {
            p = qmod(MOD - inv * temp[j][i] % MOD);
            for (k = i; k <= m; k++)
                temp[j][k] = (temp[j][k] + p * temp[i][k]) % MOD;
        }
    }
    return res;
}

void mul(Matrix a, Matrix b, Matrix c) {
    static Matrix res;
    int i, j, k;

    for (i = 1; i <= m; i++) {
        for (j = 1; j <= m; j++) {
            res[i][j] = 0;
            for (k = 1; k <= m; k++)
                res[i][j] = (res[i][j] + 1ll * b[i][k] * c[k][j]) % MOD;
        }
    }
    memcpy(a, res, sizeof(Matrix));
}

void mul(Expr a, Expr b, Expr c, Expr mod = modexpr) {
    static Expr res;
    int i, j, x;

    memset(res, 0, sizeof(Expr));
    for (i = 0; i < m; i++)
        for (j = 0; j < m; j++)
            res[i + j] = (res[i + j] + 1ll * b[i] * c[j]) % MOD;
    for (i = (m - 1) << 1; i >= m; i--) {
        if (res[i]) {
            for (j = 1; j <= m; j++)
                res[i - j] = (res[i - j] + 1ll * res[i] * mod[m - j]) % MOD;
            res[i] = 0;
        }
    }
    memcpy(a, res, sizeof(Expr));
}

    static Matrix tempmat, ansmat;
    int i, j, k;

    scanf("%s %d", s + 1, &m);
    for (i = 1; i <= m; i++)
        for (j = 1; j <= m; j++)
            scanf("%d", &mat[i][j]);

    for (i = 1; i <= m + 1; i++) {
        for (j = 1; j <= m; j++)
            mat[j][j] = qmod(MOD + mat[j][j] - 1);
        tempmat[i][1] = 1;
        for (j = 2; j <= m + 1; j++)
            tempmat[i][j] = 1ll * tempmat[i][j - 1] * i % MOD;
        tempmat[i][m + 2] = det(mat);
    }
    for (i = 1; i <= m; i++)
        mat[i][i] = qmod(mat[i][i] + m + 1);

    solve(tempmat);
    for (i = 0; i <= m; i++)
        modexpr[i] = qmod(MOD - tempmat[i + 1][m + 2]);

    b[1] = 1;
    res[0] = 1;
    for (i = strlen(s + 1); i; i--) {
        if (s[i] & 1)
            mul(res, res, b);
        mul(b, b, b);
    }

    memset(tempmat, 0, sizeof(Matrix));
    for (i = 1; i <= m; i++)
        tempmat[i][i] = 1;
    for (i = 0; i < m; i++) {
        for (j = 1; j <= m; j++)
            for (k = 1; k <= m; k++)
                ansmat[j][k] = (ansmat[j][k] + 1ll * res[i] * tempmat[j][k]) % MOD;
        mul(tempmat, tempmat, mat);
    }

    for (i = 1; i <= m; i++) {
        for (j = 1; j <= m; j++)
            printf("%d ", ansmat[i][j]);
        putchar('\n');
    }
    return 0;
}
