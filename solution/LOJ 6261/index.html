LOJ #6261. 一个人的高三楼

题目描述

一天的学习快要结束了，高三楼在晚自习的时候恢复了宁静。

不过，HSD 桑还有一些作业没有完成，他需要在这个晚自习写完。比如这道数学题：

1、给出一个数列，求它的前 $i$ 项和 $S_i$，$i\in \{x|1\le x\le n,x\in \mathbb{N}\}$

HSD 桑擅长数学，很快就把这题秒了……

然而还有第二题：

2、如果把上一问的前 $i$ 项和看成一个新数列，请求出它的前 $i$ 项和

看完第二题，还有第三题……HSD 桑已经预感到情况不妙了。

HSD 桑大致看了看题，发现有些规律。其实就是在求 $k$ 次前缀和。如果我们借用函数迭代的标记，就是在求 $S_n^{(k)}$……

HSD 桑还有很多作业要写，请你帮助他完成这项作业。

输入格式

第一行，两个正整数 $n,k$，$n$ 表示数列的长度，$k$ 的意义如题目描述；

第二行，$n$ 个正整数，表示这个数列，两个数之间用一个空格隔开。

输出格式

共 $n$ 行，每行一个数，第 $i$ 行表示 $S_i^{(k)}$，结果可能会非常大，请对 $998244353$ 取模后输出。

样例

样例输入 1

4 1
1 2 3 4

样例输出 1

1
3
6
10

样例解释 1

对于这个序列，求它的 $1$ 次前缀和，就是输出这个数列的前缀和咯……

样例输入 2

4 3
1 2 3 4

样例输出 2

1
5
15
35

样例解释 2

要求这个数列的 $3$ 次前缀和，这个数列的 $1$ 次前缀和为 $\{1,3,6,10\}$，$2$ 次前缀和为 $\{1,4,10,20\}$，$3$ 次前缀和即为 $\{1,5,15,35\}$。

见「附加文件」中 one_sample3.in

样例输出 3

见「附加文件」中 one_sample3.ans

数据范围与提示

本题为捆绑测评，只有一组子任务内所有测试点均通过才能获得该子任务的分数。

Subtask 分数 $n$ $k$
$1$ $20$ $\le 10^3$ $\le 10^2$
$2$ $30$ $\le 10^5$ $\le 2^{18}$
$3$ $50$ $\le 10^5$ $\le 2^{60}$

对于全部数据，保证数列中的任意一个数的范围在 $[1,998244353)$ 内。

题解

高次前缀和的模板题。

考虑元素 $a_i$ 对于前缀和 $S_j$ 的贡献（$i < j$）。

显然，我们可以去枚举它每一步跳到哪个地方，然后计算这个东西。

举个例子，$a_1$ 对 $S_4^{(2)}$ 的贡献有：

  • $a_1$ 被算到 $S_1^{(1)}$ 中，然后被算到 $S_4^{(2)}$ 中；
  • $a_1$ 被算到 $S_2^{(1)}$ 中，然后被算到 $S_4^{(2)}$ 中；
  • $a_1$ 被算到 $S_3^{(1)}$ 中，然后被算到 $S_4^{(2)}$ 中；
  • $a_1$ 被算到 $S_4^{(1)}$ 中，然后被算到 $S_4^{(2)}$ 中。

类似地，$a_1$ 对 $S_8^{(4)}$ 的贡献次数为

$$\sum_{i = 1}^8 \sum_{j = i}^8 \sum_{k = j}^8 1$$

本质上是枚举下一步跳到哪里去。

显然，这样的次数为 $\mathrm{C}^{k - 1}_{k - 1 + i - j}$。

也即是说

$$S_i^{(k)} = \sum_{j = 0}^i \mathrm{C}^{k - 1}_{k - 1 + i - j} a_j$$

如果我们令 $b_i = \mathrm{C}^{k - 1}_{k - 1 + i}$，那么有

$$S_i^{(k)} = \sum_{j + k = 0} b_j a_k$$

也就是一个卷积了。

DFT 优化即可。

时间复杂度：$O(n \log n)$。

代码

#include <algorithm>
#include <cstdio>

typedef long long int64;

const int MOD = 998244353;
const int G = 3;
const int INVG = 332748118;
const int MAXLOG = 20;
const int MAXN = 1 << MAXLOG;

int n, len;
int64 m;
int a[MAXN], b[MAXN];

int rev[MAXN];
int INV[MAXN];
int GT[MAXLOG + 2], INVGT[MAXLOG + 2];

inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
inline int qpow(int64 b, int p, int mod) {
    int64 res = 1;

    while (p) {
        if (p & 1)
            res = res * b % mod;
        b = b * b % mod;
        p >>= 1;
    }
    return res;
}

inline void init() {
    int i;
    int64 x;

    for (i = 1; i <= MAXLOG; i++) {
        GT[i] = qpow(G, (MOD - 1) >> i, MOD);
        INVGT[i] = qpow(INVG, (MOD - 1) >> i, MOD);
    }

    x = 1;
    for (i = 2; i < MAXN; i++)
        x = x * i % MOD;
    INV[MAXN - 1] = qpow(x, MOD - 2, MOD);
    for (i = MAXN - 2; i; i--)
        INV[i] = INV[i + 1] * (i + 1ll) % MOD;
    x = 1;
    for (i = 2; i < MAXN; x = x * i % MOD, i++)
        INV[i] = INV[i] * x % MOD;
}
inline void initrev() {
    int i, k;

    k = 1;
    while (len >> k >> 1)
        k++;
    for (i = 1; i < len; i++)
        rev[i] = (rev[i >> 1] >> 1) | ((i & 1) << (k - 1));
}
inline void dft(int a[], int len, bool inv = false) {
    int i, j, k, logk, x, y;
    int64 g, gn;

    for (i = 1; i < len; i++)
        if (i < rev[i])
            std::swap(a[i], a[rev[i]]);
    for (k = logk = 1; k < len; k <<= 1, logk++) {
        gn = inv ? INVGT[logk] : GT[logk];
        for (i = 0; i < len; i += k << 1) {
            g = 1;
            for (j = 0; j < k; j++) {
                x = a[i + j];
                y = a[i + j + k] * g % MOD;
                a[i + j] = qmod(x + y);
                a[i + j + k] = qmod(MOD + x - y);
                g = g * gn % MOD;
            }
        }
    }
    if (inv) {
        g = qpow(len, MOD - 2, MOD);
        for (i = 0; i < len; i++)
            a[i] = a[i] * g % MOD;
    }
}
#define idft(a, len) dft(a, len, true)

int main() {
#ifndef ONLINE_JUDGE
    freopen("6261.in", "r", stdin);
    freopen("6261.out", "w", stdout);
#endif

    int i;

    init();

    scanf("%d %lld", &n, &m);
    m--;
    m %= MOD;
    len = 1;
    while (len < (n << 1))
        len <<= 1;
    for (i = 0; i < n; i++)
        scanf("%d", &a[i]);
    b[0] = 1;
    for (i = 1; i < n; i++)
        b[i] = b[i - 1] * (m + i) % MOD * INV[i] % MOD;

    initrev();
    dft(a, len);
    dft(b, len);
    for (i = 0; i < len; i++)
        a[i] = 1ll * a[i] * b[i] % MOD;
    idft(a, len);
    for (i = 0; i < n; i++)
        printf("%d\n", a[i]);
    return 0;
}
