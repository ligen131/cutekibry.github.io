LOJ #2542. 「PKUWC2018」随机游走

内存限制：512 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

给定一棵 $n$ 个结点的树，你从点 $x$ 出发，每次等概率随机选择一条与所在点相邻的边走过去。

有 $Q$ 次询问，每次询问给定一个集合 $S$，求如果从 $x$ 出发一直随机游走，直到点集 $S$ 中所有点都至少经过一次的话，期望游走几步。

特别地，点 $x$（即起点）视为一开始就被经过了一次。

答案对 $998244353$ 取模。

输入格式

第一行三个正整数 $n,Q,x$。

接下来 $n-1$ 行，每行两个正整数 $(u,v)$ 描述一条树边。

接下来 $Q$ 行，每行第一个数 $k$ 表示集合大小，接下来 $k$ 个互不相同的数表示集合 $S$。

输出格式

输出 $Q$ 行，每行一个非负整数表示答案。

样例

样例输入

3 5 1
1 2
2 3
1 1
1 3
2 2 3
3 1 2 3
2 1 2
样例输出
0
4
4
4
1

样例解释

样例给的树是一条长度为 $3$ 的链，且起点是链的一端，所以答案只跟最远的那个点有关。

当最远的点是 $2$ 时，显然只需要 $1$ 步就行了。

当最远的点是 $3$ 时，通过计算可得期望 $4$ 步到达。

数据范围与提示

对于 $20\%$ 的数据，有 $1\leq n,Q\leq 5$。

另有 $10\%$ 的数据，满足给定的树是一条链。

另有 $10\%$ 的数据，满足对于所有询问有 $k=1$。

另有 $30\%$ 的数据，满足 $1\leq n\leq 10 ,Q=1$。

对于 $100\%$ 的数据，有 $1\leq n\leq 18$，$1\leq Q\leq 5000$，$1\leq k\leq n$。

题解

裸题。

显然可以做一遍 Min-Max 容斥。

设 $\max\{S\}$ 表示点集 $S$ 中所有点经过至少一次的期望步数，$\min\{S\}$ 表示走到点集 $S$ 中任意一个点的期望步数，那么有

$$\max\{S\} = \sum_{T \subseteq S} (-1)^{|T| - 1} \min\{S\}$$

由于多组询问的存在，直接枚举 $O(Q2^n)$ 明显超时。做一遍 FWT，直接得到所有 $\max\{S\}$，时间复杂度为 $O(n2^n + Q)$。

现在问题是如何求出 $\min\{S\}$。

可以设 $x_u$ 表示从点 $u$ 出发，走到点 $S$ 中任何一个点的期望步数。

那么对于 $u \in S$，有 $x_u = 0$。

对于 $u \notin S$，有

$$x_u = 1 + \sum_{u \leftrightarrow v} \frac {x_v}{deg_u}$$

$deg_u$ 为点 $u$ 度数。

按照类似于 HDU 4035 Maze（原题 / 题解）的待定系数法，可以在 $O(n)$ 时间内求解。推导不再赘述，留作读者练习。

然后可以枚举所有的 $S$，$\min\{S\} = x_1 + 1$。加上求解逆元，时间复杂度为 $O(n2^n \log 998244353)$。

时间复杂度：$O(n2^n \log 998244353 + Q)$。

代码

#include <cstdio>
#include <cstring>

typedef long long int64;

const int MOD = 998244353;
const int G = 3;
const int INVG = 332748118;

const int MAXN = 18 + 2;

int n, q, rt;
int pre[MAXN << 1], to[MAXN << 1], head[MAXN], wcnt;
int deg[MAXN], invdeg[MAXN], fa[MAXN];

bool vis[MAXN];
int a[MAXN], b[MAXN];

int ans[1 << MAXN];
int len;

inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
inline int qpow(int64 b, int p, int mod = MOD) {
    int res = 1;

    while (p) {
        if (p & 1)
            res = res * b % mod;
        b = b * b % mod;
        p >>= 1;
    }
    return res;
}

inline void addedge(int u, int v) {
    wcnt++;
    pre[wcnt] = head[u];
    head[u] = wcnt;
    to[wcnt] = v;
}
inline void add2edge(int u, int v) {
    addedge(u, v);
    addedge(v, u);
}

#define v to[i]
void predfs(int u) {
    int i;

    for (i = head[u]; i; i = pre[i]) {
        if (fa[u] ^ v) {
            fa[v] = u;
            predfs(v);
        }
    }
}
void dfs(int u) {
    int i;
    int64 p = 0;

    if (vis[u])
        return;
    vis[u] = true;

    for (i = head[u]; i; i = pre[i]) {
        if (fa[u] ^ v) {
            dfs(v);
            p = qmod(p + a[v]);
            b[u] = qmod(b[u] + b[v]);
        }
    }
    p = qpow(1 + qmod(MOD - p * invdeg[u] % MOD), MOD - 2);
    a[u] = invdeg[u] * p % MOD;
    b[u] = (1 + 1ll * b[u] * invdeg[u] % MOD) * p % MOD;
}
#undef v

int main() {
#ifndef ONLINE_JUDGE
    freopen("2542.in", "r", stdin);
    freopen("2542.out", "w", stdout);
#endif

    int i, j, k, x;
    int u, v;

    scanf("%d %d %d", &n, &q, &rt);
    for (i = 1; i < n; i++) {
        scanf("%d %d", &u, &v);
        add2edge(u, v);
        deg[u]++;
        deg[v]++;
    }
    for (i = 1; i <= n; i++)
        invdeg[i] = qpow(deg[i], MOD - 2);
    predfs(rt);

    len = 1 << n;
    for (i = 1; i < len; i++) {
        memset(a, 0, sizeof(a));
        memset(b, 0, sizeof(b));

        x = 0;
        for (j = 1; j <= n; j++) {
            vis[j] = i >> (j - 1) & 1;
            x += vis[j];
        }
        dfs(rt);
        ans[i] = b[rt];
        if (!(x & 1) and ans[i])
            ans[i] = MOD - ans[i];
    }

    for (k = 1; k < len; k <<= 1)
        for (i = 0; i < len; i += k << 1)
            for (j = 0; j < k; j++)
                ans[i + j + k] = qmod(ans[i + j + k] + ans[i + j]);

    while (q) {
        u = 0;
        scanf("%d", &k);
        while (k) {
            scanf("%d", &x);
            u |= 1 << (x - 1);
            k--;
        }
        printf("%d\n", ans[u]);
        q--;
    }
    return 0;
}
