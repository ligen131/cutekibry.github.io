LOJ #6272. 「BestCoder Round #88」Tree Cutting（改）

内存限制：256 MiB 时间限制：1000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

给一棵 $n(1 \le n \le 1000)$ 个节点的树，每个点有点权 $w_i(0 \le w_i < 2 ^ {15})$ 。

设 $f_x$ 表示树上有多少个连通块，满足点权异或和为 $x$ 。

求 $f_0, f_1, f_2, ..., f_{2 ^ {15} - 1}$ 对 $998244353$ 取模。

输入格式

第一行一个正整数 $n$ 。

第二行 $n$ 个非负整数 $w_i$ 。

接下来 $n-1$ 行，每行两个正整数 $x, y$ ，表示树上存在边 $(x, y)$ 。

输出格式

输出一行，共 $2 ^ {15}$ 个非负整数，依次为 $f_0, f_1, ..., f_{2 ^ {15} - 1}$ 。

数据范围与提示

对于 $100\%$ 的数据，$1 \le n \le 1000, 0 \le w_i <2 ^ {15}$ 。

题解

用 FWT 维护树形 DP。

#define v to[i]
void dfs(int u, int fa) {
    int i, j;

    a[u][val[u]] = 1;
    fwt(a[u]);
    for (i = head[u]; i; i = pre[i]) {
        if (fa ^ v) {
            dfs(v, u);
            for (j = 0; !(j>> LOG); j++)
                a[u][j] = (int64)a[u][j] * (a[v][j] + 1) % MOD;
        }
    }
}
#undef v

dfs(1, 0);

这样直接做是 $O(n w_i \log w_i)$ 的，虽然卡常也可能能通过此题，但不是最优的。

瓶颈出现在对每个节点的 FWT 上。如果我们能把每次 FWT 的复杂度降到 $O(w_i)$，那么就可以在 $O(nw_i + w_i \log w_i)$ 时间内通过此题了。

x = a[i + j];
y = a[i + j + k];
a[i + j] = x + y;
a[i + j + k] = x - y;

我们 FWT 的时候，实际上进行了一些无用的操作。如果 $a[i + j] = a[i + j + k] = 0$，那么这个操作实际上是没有效果的。

XHRlyb 上传的 最优解 给出了一个 $O(w_i)$ 的 FWT 函数：

void init(int f[], int w, int l, int r) {
    if (l + 1 == r) {
        f[l] = 1;
        return;
    }
    int mid = l + r >> 1;
    if (w < mid) {
        init(f, w, l, mid);
        for (int i = mid, j = l; i < r; i++, j++)
            f[i] = f[j];
    }
    else {
        init(f, w, mid, r);
        for (int i = l, j = mid; i < mid; i++, j++)
            f[i] = f[j], f[j] = sub(0, f[j]);
    }
}

在这里，$l, r$ 代表对 $[l, r]$ 进行一次 FWT 迭代。

很明显地，如果 $a[l \ldots mid] = 0$，那么对这一部分就没有必要进行 FWT 了。根据这一点，就可以写出 $O(w_i)$ 的 FWT 代码。

注意：这个代码只能适用于对于 $A_p=1, A_i=0 \: (i \neq p)$ 序列的 FWT，对于通常的 FWT 仍然需要 $O(w_i \log w_i)$ 的复杂度进行计算。

代码

#include <cstdio>

typedef long long int64;

const int MAXN = 1000 + 10;
const int LOG = 15;
const int MOD = 998244353;
const int INVLEN = 998213889;

int a[MAXN][1 << LOG];
int pre[MAXN << 1], to[MAXN << 1], head[MAXN], wcnt;
int val[MAXN];
int ans[1 << LOG];
int n;

inline void read(int &x) {
    char ch;

    do
        ch = getchar();
    while (ch < '0' or ch> '9');
    x = 0;
    while (ch>= '0' and ch <= '9') {
        x = (x << 3) + (x << 1) + (ch & 15);
        ch = getchar();
    }
}
inline void read(int &a, int &b) {
    read(a);
    read(b);
}
inline void write(int x) {
    static char buf[20];
    int top = 0;

    if (!x) {
        putchar('0');
        return;
    }
    while (x) {
        top++;
        buf[top] = (x % 10) | 48;
        x /= 10;
    }
    while (top) {
        putchar(buf[top]);
        top--;
    }
}
inline void write_(int x) {
    write(x);
    putchar(' ');
}
inline int qmod(int x) {
    while (x < 0)
        x += MOD;
    while (x>= MOD)
        x -= MOD;
    return x;
}
inline void addedge(int u, int v) {
    wcnt++;
    pre[wcnt] = head[u];
    head[u] = wcnt;
    to[wcnt] = v;
}
inline void add2edge(int u, int v) {
    addedge(u, v);
    addedge(v, u);
}

#define mid ((l + r) >> 1)
inline void init(int *a, int l, int r, int x) {
    int i, j;

    if (l == r - 1) {
        a[l] = 1;
        return;
    }
    else if (x < mid) {
        init(a, l, mid, x);
        for (i = l, j = mid; j < r; i++, j++)
            a[j] = a[i];
    }
    else {
        init(a, mid, r, x);
        for (i = l, j = mid; j < r; i++, j++) {
            a[i] = a[j];
            a[j] = qmod(-a[j]);
        }
    }
}
#undef mid
inline void fwt(int *a) {
    int i, j, k, x, y;

    for (k = 1; !(k>> LOG); k <<= 1) {
        for (i = 0; !(i>> LOG); i += k << 1) {
            for (j = 0; j < k; j++) {
                x = a[i + j];
                y = a[i + j + k];
                a[i + j] = qmod(x + y);
                a[i + j + k] = qmod(x - y);
            }
        }
    }
}
inline void ifwt(int *a) {
    int i;

    fwt(a);
    for (i = 0; !(i>> LOG); i++)
        a[i] = (int64)a[i] * INVLEN % MOD;
}

#define v to[i]
void dfs(int u, int fa) {
    int i, j;

    init(a[u], 0, 1 << LOG, val[u]);
    for (i = head[u]; i; i = pre[i]) {
        if (fa ^ v) {
            dfs(v, u);
            for (j = 0; !(j>> LOG); j++)
                a[u][j] = (int64)a[u][j] * (a[v][j] + 1) % MOD;
        }
    }
}
#undef v

int main() {
#ifndef ONLINE_JUDGE
    freopen("6272.in", "r", stdin);
    freopen("6272.out", "w", stdout);
#endif
    int i, j, x, y;

    read(n);
    for (i = 1; i <= n; i++)
        read(val[i]);
    for (i = 1; i < n; i++) {
        read(x, y);
        add2edge(x, y);
    }

    dfs(1, 0);
    for (i = 1; i <= n; i++)
        for (j = 0; !(j>> LOG); j++)
            ans[j] = qmod(ans[j] + a[i][j]);
    ifwt(ans);

    for (i = 0; !(i>> LOG); i++)
        write_(ans[i]);
    return 0;
}
