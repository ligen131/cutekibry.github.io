LOJ #6539. 奇妙数论题

内存限制：512 MiB 时间限制：1500 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

小 $Z$ 在学习数论的时候想到一个这样的问题。

给你一个长为 $n$ 的 排列 $a$ ，求

$$ \sum_{i = 1}^{n} \sum_{j = 1}^{n} (a_i, a_j) $$

注意 $(a_i, a_j)$ 表示 $\gcd(a_i, a_j)$ ，也就是 $a_i, a_j$ 的最大公约数。

小 $X$ 看了一眼，说道这不是道裸题吗，然后把小 $Z$ 吊了一顿。

于是小 $Z$ 想了想如何加强，于是多加了一个乘数。

给你一个长为 $n$ 的 排列 $a$ ，求

$$ \sum_{i = 1}^{n} \sum_{j = 1}^{n} (a_i, a_j) \cdot (i, j) $$

小 $X$ 又认真看了看，又说“有什么区别”。

小 $Z$ 没有办法加强了，十分无奈地把这题丢了出来给聪明的你做。

输入格式

第一行两个正整数 $n$ 。

接下来一行共 $n$ 个正整数，其中第 $i$ 个表示 $a_i$ 。

输出格式

输出一行一个数，表示答案，对于 $10^9 + 7$ 取模。

样例

输入

5
1 4 5 2 3

输出

73

数据范围与提示

保证 $n \le 10^5$ 且 $a_i$ 为一个 排列 。

题解

显然可以莫比乌斯反演。

套路的反演：

$$ \begin{aligned} & \sum_{i = 1}^n \sum_{j = 1}^n \gcd(a_i, a_j) \times \gcd(i, j) \\ =&\sum_{d = 1}^n d \sum_{i = 1}^n \sum_{j = 1}^n \gcd(a_i, a_j) [\gcd(i, j) = d] \\ =&\sum_{d = 1}^n d \sum_{d|x} \mu(\frac xd) \sum_{i = 1}^n \sum_{j = 1}^n \gcd(a_i, a_j) [x | \gcd(i, j)] \\ =&\sum_{x = 1}^n \sum_{i = 1}^n \sum_{j = 1}^n \gcd(a_i, a_j) [x | \gcd(i, j)] \sum_{d | x} d \mu(\frac xd) \\ =&\sum_{x = 1}^n \varphi(x) \sum_{i = 1}^n \sum_{j = 1}^n \gcd(a_i, a_j) [x | \gcd(i, j)]\\ =&\sum_{x = 1}^n \varphi(x) \sum_{i = 1}^{\lfloor \frac nx \rfloor} \sum_{j = 1}^{\lfloor \frac nx \rfloor} \gcd(a_{ix}, a_{jx}) \\ =&\sum_{x = 1}^n \varphi(x) \sum_{y = 1}^n \varphi(y) \sum_{i = 1}^{\lfloor \frac nx \rfloor} \sum_{j = 1}^{\lfloor \frac nx \rfloor} [y|\gcd(a_{ix}, a_{jx})] \\ \end{aligned} $$

枚举 $x$，对后面那个东西维护一个计数器即可。

因为 $2 \times 3 \times 5 \times 7 \times 11 \times 13 \times 17 = 510510 > 100000$，所以 $100000$ 以内的数拥有不超过 $2^7 = 128$ 个因子。

时间复杂度：$O(n \log^2 n)$。

经孔爷指证，复杂度是 $O(n \log^3 n)$ 的。

（我也不知道为什么是 $O(n \log^3 n)$ 的）

代码

#include <algorithm>
#include <cstdio>

const int MAXN = 100000 + 10;
const int MOD = 1000000007;

int n;
int a[MAXN];
int d[MAXN][129];

int stk[MAXN], stop;
int sum[MAXN];
int phi[MAXN];

void linear_sleve() {
    static int p[MAXN];
    static bool notp[MAXN];
    int i, j, pn = 0;

    phi[1] = 1;
    for (i = 2; i <= n; i++) {
        if (!notp[i]) {
            pn++;
            p[pn] = i;
            phi[i] = i - 1;
        }
        for (j = 1; j <= pn and i * p[j] <= n; j++) {
            notp[i * p[j]] = true;
            if (i % p[j])
                phi[i * p[j]] = phi[i] * phi[p[j]];
            else {
                phi[i * p[j]] = phi[i] * p[j];
                break;
            }
        }
    }
}

void getdiv(int d[], int x) {
    int i;

    for (i = 1; i * i <= x; i++) {
        if (!(x % i)) {
            d[0]++;
            d[d[0]] = i;
            if (i * i != x) {
                d[0]++;
                d[d[0]] = x / i;
            }
        }
    }
}
int F(int k) {
    int i, j;
    int ans = 0;

    for (i = k; i <= n; i += k) {
        for (j = 1; j <= d[i][0]; j++) {
            if (!sum[d[i][j]]) {
                stop++;
                stk[stop] = d[i][j];
            }
            sum[d[i][j]]++;
        }
    }
    while (stop) {
        ans = (ans + 1ll * sum[stk[stop]] * sum[stk[stop]] % MOD * phi[stk[stop]]) % MOD;
        sum[stk[stop]] = 0;
        stop--;
    }
    return ans;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("6539.in", "r", stdin);
    freopen("6539.out", "w", stdout);
#endif

    int i;
    int ans = 0;

    scanf("%d", &n);
    for (i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
        getdiv(d[i], a[i]);
    }

    linear_sleve();
    for (i = 1; i <= n; i++)
        ans = (ans + 1ll * phi[i] * F(i)) % MOD;
    printf("%d\n", ans);
    return 0;
}
