LOJ #2409. 「THUPC 2017」小 L 的计算题 / Sum

内存限制：512 MiB 时间限制：3000 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

现有一个长度为 $n$ $(1 \le n\le 2\times 10^5)$ 的非负整数数组 $\{a_i\}$。小 L 定义了一种神奇变换：

$$ f_k=\sum_{i=1}^{n}{a_i}^k\pmod{ 998244353 } $$

小 L 计划用变换生成的序列 $f$ 做一些有趣的事情，但是他并不擅长算乘法，所以来找你帮忙，希望你能帮他尽快计算出 $f_1\sim f_n$。

输入格式

输入包含多组数据。

输入的第一行包含一个整数 $T$ $(1\le T\le 20)$, 表示数据组数。

接下来 $2T$ 行，每两行代表一组测试数据。

每组数据的第一行包含一个整数 $n$，表示数组 $\{a_i\}$ 的长度。接下来一行 $n$ 个整数，描述数组 $\{a_i\}$。

保证输入的 $a_i$ 满足 $0\le a_i\le 10^9$。在一个测试文件中，保证有 $\sum n\le 4\times 10^5$。

输出格式

我们不想让你输出过多的数。因此，令 $\text{ans}=f_1\oplus f_2\oplus\dots\oplus f_n$，其中 $\oplus$ 表示异或操作，在 C++ / Java / Python 中，它可以表示为 ^。

对每组数据，你需要输出一行一个整数，表示 $\text{ans}$。

样例

样例输入

2
3
2 3 3
5
1 2 3 4 5

样例输出

32
4675

题解

首先，我们有

$$\ln(1 + x) = \sum_{i = 1}^n \frac {(-1)^{i + 1}}i x^i + O(x^n)$$

注意到这里有个 $x^i$ 的求和项，因此我们考虑用 $\ln$ 来转化这个东西。

设生成函数 $F(x) = \prod_{i = 1}^n (1 + a_ix)$，那么有

$$ \begin{aligned} \ln F(x) &= \ln \prod_{i = 1}^n (1 + a_ix) \\ &= \sum_{i = 1}^n \ln(1 + a_ix) \\ &= \sum_{i = 1}^n \sum_{k = 1}^\infty \frac {(-1)^{k + 1}}k a_i^kx^k \\ &= \sum_{k = 1}^\infty \sum_{i = 1}^n \frac {(-1)^{k + 1}}k a_i^kx^k \\ &= \sum_{k = 1}^\infty \frac {(-1)^{k + 1}}k \sum_{i = 1}^n a_i^kx^k \\ \end{aligned} $$

我们可以用分治 DFT $O(n \log^2 n)$ 求出生成函数 $F(x)$，然后用 $O(n \log n)$ 的多项式 $\ln$ 求出 $\ln F(x)$。

时间复杂度：$O(Tn \log^2 n)$。

代码

#include <algorithm>
#include <cstdio>
#include <cstring>

const int MAXLOG = 20;
const int MAXN = 1 << MAXLOG;
const int MOD = 998244353;
const int G = 3;
const int INVG = 332748118;

typedef long long int64;
typedef int Expr[MAXN];

int n, len;
int a[MAXN];
int rev[MAXN];

int INV[MAXN];
int GT[MAXLOG + 1], INVGT[MAXLOG + 1];

int f[MAXN];

inline void read(int &x) {
    char ch;

    do
        ch = getchar();
    while (ch < '0' or ch > '9');
    x = 0;
    while (ch >= '0' and ch <= '9') {
        x = (x << 3) + (x << 1) + (ch & 15);
        ch = getchar();
    }
}
inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
inline int qpow(int64 b, int p, int mod) {
    int res = 1;
    while (p) {
        if (p & 1)
            res = res * b % mod;
        b = b * b % mod;
        p >>= 1;
    }
    return res;
}

inline void init() {
    int i;
    int64 x;

    x = 1;
    for (i = 2; i < MAXN; i++)
        x = x * i % MOD;
    x = qpow(x, MOD - 2, MOD);
    INV[MAXN - 1] = x;
    for (i = MAXN - 2; i; i--)
        INV[i] = INV[i + 1] * (i + 1ll) % MOD;
    x = 1;
    for (i = 2; i < MAXN; i++) {
        INV[i] = x * INV[i] % MOD;
        x = x * i % MOD;
    }

    for (i = 1; i <= MAXLOG; i++) {
        GT[i] = qpow(G, (MOD - 1) / (1 << i), MOD);
        INVGT[i] = qpow(INVG, (MOD - 1) / (1 << i), MOD);
    }
}
inline void initrev(int len) {
    int i, k;
    k = 1;
    while (len >> k >> 1)
        k++;
    for (i = 1; i < len; i++)
        rev[i] = (rev[i >> 1] >> 1) | ((i & 1) << (k - 1));
}
inline void dft(Expr a, int len, bool inv = false) {
    int i, j, k, logk;
    int x, y;
    int64 g, gn;

    for (i = 1; i < len; i++)
        if (i < rev[i])
            std::swap(a[i], a[rev[i]]);
    for (k = logk = 1; k < len; k <<= 1, logk++) {
        gn = inv ? INVGT[logk] : GT[logk];
        for (i = 0; i < len; i += k << 1) {
            g = 1;
            for (j = 0; j < k; j++) {
                x = a[i + j];
                y = g * a[i + j + k] % MOD;
                a[i + j] = qmod(x + y);
                a[i + j + k] = qmod(x - y + MOD);
                g = g * gn % MOD;
            }
        }
    }
    if (inv) {
        x = qpow(len, MOD - 2, MOD);
        for (i = 0; i < len; i++)
            a[i] = 1ll * a[i] * x % MOD;
    }
}
inline void idft(Expr a, int len) { dft(a, len, true); }

void solve(int len) {
    int i, j, k;
    static Expr b;

    for (i = 0; i < len; i++) {
        f[i << 1] = 1;
        f[i << 1 | 1] = a[i];
    }
    len <<= 1;
    for (k = 2; k < len; k <<= 1) {
        initrev(k << 1);
        for (i = 0; i < len; i += k << 1) {
            memcpy(b, f + i + k, k << 2);
            memset(b + k, 0, k << 2);
            memset(f + i + k, 0, k << 2);
            dft(f + i, k << 1);
            dft(b, k << 1);
            for (j = 0; j < k << 1; j++)
                f[i + j] = 1ll * f[i + j] * b[j] % MOD;
            idft(f + i, k << 1);
        }
    }
    memset(b, 0, sizeof(Expr));
}
void inv(Expr res, Expr b, int len) { //
    static Expr a, bb;
    int i, k;

    a[0] = qpow(b[0], MOD - 2, MOD);
    for (k = 2; k <= (len << 1); k <<= 1) {
        initrev(k);
        memcpy(bb, b, k << 1);
        dft(a, k);
        dft(bb, k);
        for (i = 0; i < k; i++)
            a[i] = 1ll * a[i] * qmod(MOD + 2 - 1ll * a[i] * bb[i] % MOD) % MOD;
        idft(a, k);
        for (i = k >> 1; i < k; i++)
            a[i] = 0;
    }
    memcpy(res, a, sizeof(Expr));
    memset(a, 0, sizeof(Expr));
    memset(bb, 0, sizeof(Expr));
}
void det(Expr a, Expr b, int len) { // a(x^p)' = apx^(p - 1)
    int i;
    for (i = 1; i < len; i++)
        a[i - 1] = 1ll * b[i] * i % MOD;
    a[len - 1] = 0;
}
void int_(Expr a, Expr b, int len) {
    int i;
    for (i = len - 1; i; i--)
        a[i] = 1ll * b[i - 1] * INV[i] % MOD;
    a[0] = 0;
}
inline void ln(Expr res, Expr b, int len) {
    static Expr a, inva;
    int i;

    len <<= 1;

    memcpy(a, b, len << 1);
    det(a, a, len >> 1);
    inv(inva, b, len >> 1);

    initrev(len);
    dft(a, len);
    dft(inva, len);
    for (i = 0; i < len; i++)
        res[i] = 1ll * a[i] * inva[i] % MOD;
    idft(res, len);

    int_(res, res, len);

    memset(a, 0, sizeof(Expr));
    memset(inva, 0, sizeof(Expr));
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("2409.in", "r", stdin);
    freopen("2409.out", "w", stdout);
#endif

    int T;
    int i, ans;
    static Expr g;

    init();

    read(T);
    while (T) {
        memset(f, 0, sizeof(f));
        memset(g, 0, sizeof(g));
        memset(a, 0, sizeof(a));

        read(n);
        len = 1;
        while (len <= n)
            len <<= 1;
        for (i = 0; i < n; i++)
            read(a[i]);

        solve(len);
        ln(g, f, len);
        ans = 0;
        for (i = 1; i <= n; i++)
            ans ^= 1ll * ((i & 1) ? g[i] : (MOD - g[i])) * i % MOD;
        printf("%d\n", ans);

        T--;
    }
    return 0;
}
