题目类型：传统 评测方式：文本比较

题目描述

小 Q 在他的个人主页上放出了一个悬赏：征集只含正整数的非空集合 $S$，其中的每个元素都不超过 $n$，并且满足一些附加条件。

众所周知，我们可以很轻松地对于任意不超过 $n$ 的正整数 $x$，计算出把 $x$ 表示成 $S$ 中元素之和的方案数 $f(x)$，在这里我们约定，在任意方案中每个数字可以出现多次，但是不考虑数字出现的顺序。

例如，当 $S = \{1,2,3,4,5\}$ 时，我们可以计算出 $f(2) = 2$，$f(3) = 3$，$f(4) = 5$，$f(5) = 7$。

再例如，当 $S = \{1,2,5\}$ 时，我们可以计算出 $f(4) = 3$，$f(5) = 4$，$f(6) = 5$，$f(7) = 6$。

麻烦地是现在小 Q 忘记了 $S$ 里有哪些元素，幸运地是他用存储设备记录下了所有 $f(i) \bmod p$ 的值，小 Q 希望你能利用这些信息帮他恢复出 $S$ 原来的样子。

具体来说，他希望你找到这样一个正整数的非空集合 $S$，其中的每个元素都不超过 $n$，并且对于任意的 $i = 1,2,··· ,n$，满足把 $i$ 表示成 $S$ 中元素之和的方案数在模 $p$ 意义下等于 $f(i)$，其中 $p$ 是记录在存储设备中的一个质数。他向你保证：一定存在这样的集合 $S$。

然而，小 Q 觉得他存储的信息并不足以恢复出唯一的 $S$，也就是说，可能会存在多个这样的集合 $S$，所以小 Q 希望你能给出所有解中字典序最小的解。

对于满足条件的两个不同的集合 $S_1$ 和 $S_2$，我们认为 $S_1$ 的字典序比 $S_2$ 的字典序小，当且仅当存在非负整数 $k$，使得 $S_1$ 的前 $k$ 小元素与 $S_2$ 的前 $k$ 小元素完全相等，并且，要么 $S_1$ 的元素个数为 $k$，且 $S_2$ 的元素个数至少为 $(k+1)$，要么 $S_1$ 和 $S_2$ 都有至少 $(k+1)$ 个元素，且 $S_1$ 的第 $(k+1)$ 小元素比 $S_2$ 的第 $(k+1)$ 小元素小。

输入格式

第一行包含两个整数 $n$ 和 $p$，满足 $p$ 是质数。

第二行包含 $n$ 个整数 $f(1), f(2), \dots , f(n)$，含义见题目描述。

保证每一行中相邻的整数由恰好一个空格隔开，并且末尾没有多余的空格。

输出格式

你需要输出两行信息来描述字典序最小的解，其中第一行包含一个整数 $m (m > 0)$，表示 $S$ 的元素个数，第二行包含 $m$ 个正整数 $s_1, s_2, \dots , s_m$，表示将 $S$ 的所有元素按升序排序后得到的序列。

你需要保证输出的每一行中相邻的整数由恰好一个空格隔开，并且每一行的末尾没有多余的空格。

样例

样例输入 1

5 1000003
1 2 3 5 7

样例输出 1

5
1 2 3 4 5

样例输入 2

9 1000003
1 2 2 3 4 5 6 7 8

样例输出 2

3

题解

基础题。

强行枚举 $1 \ldots n$ 的物品有没有出现，可以在 $O(n^2)$ 时间内解决问题。通过这个枚举过程，我们可以发现方案唯一，因此不需要考虑字典序问题。

考虑一般型生成函数。

令 $s_0 = 1$。设 $F(x) = \sum_{i = 0}^n s_i x^i$，那么已知 $s_1, s_2, \ldots, s_n$，求将 $F(x)$ 分解为若干 $\frac 1{1 - x^k} \pmod {x^{n + 1}}$ 的乘积。

数据保证 $F(x)$ 可以分解。

以下省略 $\bmod \: x^{n + 1}$。

$$F(x) = \prod_{i = 1}^n (\frac 1{1 - x^i})^{a_i}$$

显然，直接做太复杂了。经验告诉我们，面对这种奇怪的乘积问题——遇事不决先 $\ln$。

$$ \begin{aligned} \ln F(x) &= \ln \prod_{i = 1}^n (\frac 1{x^i - 1})^{a_i} \\ &= \sum_{i = 1}^n a_i \ln (\frac 1{1 - x^i}) \\ &= -\sum_{i = 1}^n a_i \ln (1 - x^i) \\ &= \sum_{i = 1}^n a_i \sum_{j = 1}^\infty \frac {x^{ij}}j \\ \end{aligned} $$

对 $F(x)$ 简单取个 $\ln$ 就可以得到右边的式子。

现在我们要求的即为 $a_i$。设 $\ln F(x)$ 的第 $i$ 项系数为 $b_i$，则

$$ \begin{aligned} b_i &= \sum_{j | i} \frac {a_j}{(\frac ij)} \\ b_i &= \sum_{j | i} \frac {ja_j}{i} \\ ib_i &= \sum_{j | i} ja_j \\ \end{aligned} $$

对这个式子做一遍莫比乌斯反演，得

$$ia_i = \sum_{j | i} \mu(\frac ij) jb_j$$

时间复杂度：$O(n \log n)$。

毒瘤出题人还要 MTT，报警了。

附：多项式逆元

$$ \begin{aligned} (B - B_0)^2 &\equiv 0 \\ B^2 - 2BB_0 + B_0^2 &\equiv 0 \\ B^2 &\equiv 2BB_0 - B_0^2 \\ B &\equiv 2B_0 - AB_0^2 \\ B &\equiv B_0 (2 - AB_0) \\ \end{aligned} $$

多项式 $\ln$

$$ \begin{aligned} B &\equiv \ln A \\ B' &\equiv (\ln A)' \\ B' &\equiv \frac {A'}{A} \\ B &\equiv \int \frac {A'}{A} \\ \end{aligned} $$

代码

#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>

typedef long long int64;
typedef long double extend;

const extend PI = std::acos(extend(-1));

#ifdef ONLINE_JUDGE
const int MAXLOG = 20;
#else
const int MAXLOG = 7;
#endif

const int MAXN = 1 << MAXLOG;

struct Complex {
    extend r, i;
    Complex() {}
    Complex(extend _r, extend _i) {
        r = _r;
        i = _i;
    }
    Complex operator=(extend x) {
        r = x;
        i = 0;
        return *this;
    }
    Complex conj() { return Complex(r, -i); }
    Complex operator+(Complex b) { return Complex(r + b.r, i + b.i); }
    Complex operator-(Complex b) { return Complex(r - b.r, i - b.i); }
    Complex operator*(Complex b) { return Complex(r * b.r - i * b.i, r * b.i + i * b.r); }
    Complex operator/(extend x) { return Complex(r / x, i / x); }
};
typedef Complex CExpr[MAXN];
typedef int IExpr[MAXN];

int n, MOD;
int rev[MAXN];
int mu[MAXN];
Complex WN[MAXLOG + 2];

inline int qpow(int64 b, int p) {
    int res = 1;
    while (p) {
        if (p & 1)
            res = res * b % MOD;
        b = b * b % MOD;
        p >>= 1;
    }
    return res;
}
inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }

// Mu
inline void linear_sleve(int n) {
    int i, j;
    static int p[MAXN], pn;
    static bool notp[MAXN];

    mu[1] = 1;
    for (i = 2; i <= n; i++) {
        if (!notp[i]) {
            pn++;
            p[pn] = i;
            mu[i] = -1;
        }
        for (j = 1; j <= pn and i * p[j] <= n; j++) {
            notp[i * p[j]] = true;
            if (i % p[j])
                mu[i * p[j]] = -mu[i];
            else {
                mu[i * p[j]] = 0;
                break;
            }
        }
    }
    for (i = 2; i <= n; i++)
        if (mu[i] == -1)
            mu[i] = MOD - 1;
}

// Expr
inline int64 roundm(extend x) { return (int64)(x + 0.5) % MOD; }
inline void initwn() {
    int i, k;
    for (k = i = 1; i <= MAXLOG; k <<= 1, i++)
        WN[i] = Complex(std::cos(PI / k), std::sin(PI / k));
}
inline void initrev(int len) {
    int i, k = 1;
    while (len >> k >> 1)
        k++;
    for (i = 1; i < len; i++)
        rev[i] = (rev[i >> 1] >> 1) | ((i & 1) << (k - 1));
}
void dft(CExpr a, int len) {
    int i, j, k, logk;
    Complex wn, w, t;

    for (i = 1; i < len; i++)
        if (i < rev[i])
            std::swap(a[i], a[rev[i]]);
    for (k = logk = 1; k < len; k <<= 1, logk++) {
        wn = WN[logk];
        for (i = 0; i < len; i += k << 1) {
            w = 1;
            for (j = 0; j < k; j++) {
                t = a[i + j + k] * w;
                a[i + j + k] = a[i + j] - t;
                a[i + j] = a[i + j] + t;
                w = w * wn;
            }
        }
    }
}
void mul(IExpr res, IExpr a, IExpr b, int len) {
    static CExpr a1, b1, c1, c2;
    int i, j;

    for (i = 0; i < len; i++) {
        a1[i] = Complex(a[i] & 32767, a[i] >> 15);
        b1[i] = Complex(b[i] & 32767, b[i] >> 15);
    }
    dft(a1, len);
    dft(b1, len);
    for (i = 0; i < len; i++) {
        j = i ? (len - i) : 0;
        c1[j] = (a1[i] + a1[j].conj()) * Complex(0.5, 0) * b1[i];
        c2[j] = (a1[i] - a1[j].conj()) * Complex(0, -0.5) * b1[i];
    }
    dft(c1, len);
    dft(c2, len);
    for (i = 0; i < len; i++) {
        c1[i] = c1[i] / len;
        c2[i] = c2[i] / len;
        res[i] =
            ((roundm(c2[i].i) << 30) + ((roundm(c2[i].r) + roundm(c1[i].i)) << 15) + roundm(c1[i].r)) % MOD;
    }
}
void inv(IExpr res, IExpr a, int len) {
    static IExpr a0, b, c;
    int i, k;

    b[0] = qpow(a[0], MOD - 2);
    for (k = 4; k <= (len << 1); k <<= 1) {
        initrev(k);

        memcpy(a0, a, k << 1);

        mul(c, a0, b, k);
        for (i = 0; i < k; i++)
            c[i] = qmod(MOD - c[i]);
        c[0] = qmod(c[0] + 2);
        mul(b, b, c, k);
        memset(b + (k >> 1), 0, k << 1);
    }
    memcpy(res, b, k << 2);
}
inline void det(IExpr a, IExpr b, int len) {
    int i;
    for (i = 1; i < len; i++)
        a[i - 1] = 1ll * b[i] * i % MOD;
    a[len - 1] = 0;
}
inline void int_(IExpr a, IExpr b, int len) {
    int i;
    for (i = len - 1; i; i--)
        a[i] = 1ll * b[i - 1] * qpow(i, MOD - 2) % MOD;
    a[0] = 0;
}
void ln(IExpr res, IExpr a, int len) {
    static IExpr deta, inva;

    det(deta, a, len);
    inv(inva, a, len);
    initrev(len << 1);
    mul(res, deta, inva, len << 1);
    int_(res, res, len << 1);
    memset(res + len, 0, len << 2);
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("2271.in", "r", stdin);
    freopen("2271.out", "w", stdout);
#endif

    static IExpr a, b;
    int i, j, len, tot = 0;

    initwn();

    scanf("%d %d", &n, &MOD);
    for (i = 1; i <= n; i++)
        scanf("%d", &b[i]);
    b[0] = 1;

    len = 1;
    while (len <= n)
        len <<= 1;

    ln(b, b, len);

    linear_sleve(len - 1);
    for (i = 1; i <= n; i++)
        b[i] = 1ll * b[i] * i % MOD;
    for (i = 1; i <= n; i++)
        for (j = 1; i * j <= n; j++)
            a[i * j] = (a[i * j] + 1ll * mu[j] * b[i]) % MOD;
    for (i = 1; i <= n; i++)
        if (a[i])
            tot++;
    printf("%d\n", tot);
    for (i = 1; i <= n; i++)
        if (a[i])
            printf("%d ", i);
    putchar('\n');

    return 0;
}
