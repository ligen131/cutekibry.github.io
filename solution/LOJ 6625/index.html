LOJ #6625. 时间复杂度

内存限制：256 MiB 时间限制：2333 ms 标准输入输出
题目类型：传统 评测方式：文本比较
上传者：未知
原题链接 查看源代码 修改历史 版本对比 ::after

题目描述

一年半前，在 nantf 是个连线性筛和 $O(\sqrt{n})$ 判断素数都不会的蒟蒻时，他遇到了这样一道题：

给定正整数 $n$，求在 $1$ 到 $n$ 之间有多少个质数。

$1\le n\le 10^{11}$。

nantf 当然知道暴力会超时，但是他也没有办法。

突然，他想到了：判断素数的复杂度是跑不满的！

于是他打了下面的暴力：（C++ 代码）

typedef long long ll;
bool check(ll x){
    if(x == 1) return false;
    for(ll i = 2; i < x; i++)
        if(x % i == 0) return false;
    return true;
}
int main(){
    ll n, ans = 0;
    cin >> n;
    for(ll i = 1; i <= n; i++)
        if(check(i)) ans++;
    cout << ans << endl;
}

虽然这个程序复杂度跑不满 $O(n^2)$，但是它还是毫无疑问地挂了。

nantf 不服气，他把 check 函数改写了一下：

typedef long long ll;
ll cnt = 0;
bool check(ll x){
    if(x == 1) return false;
    for(ll i = 2; i < x; i++){
        cnt++;
        if(x % i == 0) return false;
    }
    return true;
}

众所周知程序结束时 $cnt$ 的值能反应程序的实际复杂度。

所以 nantf 找到了你，想让你帮忙算一下程序结束时 $cnt$ 的值。不需要考虑溢出。

以上是一年半前 nantf 的惨痛回忆。

现在 nantf 已经会做这题了，但他现在对这个 $cnt$ 很感兴趣，想要计算它的值。

nantf 当然知道怎么做啦！但是他想考考你……

另外，nantf 已经不关心这个程序的时间复杂度了，于是他决定让你求出 $cnt$ 对 $20190601$（一个质数）取模的值。

输入格式

仅一行，包含一个正整数 $n$。

输出格式

仅一行，输出 $cnt$ 对 $20190601$ 取模的值。不需要考虑溢出。

样例

样例输入1

10

样例输出1

15

样例输入2

20190601

样例输出2

10608902

数据范围与提示

对于 $10\%$ 的数据，$1\le n\le 10^2$。

对于 $20\%$ 的数据，$1\le n\le 10^3$。

对于 $40\%$ 的数据，$1\le n\le 10^7$。

对于 $70\%$ 的数据，$1\le n\le 10^9$。

对于 $100\%$ 的数据，$1\le n\le 10^{11}$。

题解

注意到 $n \leq 10^{11}$，可以考虑洲阁筛。

设 $minp(i)$ 为 $i$ 的最小质因子，则答案等于

$$\left( \sum_{i = 1}^n minp(i) \right) - n - \left( \sum_{i = 1}^n [i \in \mathbb{P}] \right)$$

后面两项显然可以直接洲阁筛求出。

考虑如何求出前面的东西。

显然，可以枚举最小的质因子 $p$ 和质因子的质数 $c$：

$$\sum_{i = 1}^n minp(i) = \sum_{i = 1}^m \sum_{c \in \mathbb{N}^+} p_i f(\lfloor \frac n{p_i^c} \rfloor, i)$$

这里 $f(n, k)$ 表示 $[1, n]$ 中与 $p_1, p_2, \ldots, p_k$ 互质的整数个数。这是很明显的，因为只有当 $x$ 的因子中没有 $p_1, p_2, \ldots, p_k$ 时，$p_k$ 才是 $p_k^cx$ 的最小质因子。

显然，这个东西其实我们在洲阁筛的过程中就已经求出来了。

因此直接洲阁筛即可，实现细节留给读者思考。

时间复杂度：$O(\frac {n^{\frac 34}}{\log n})$。

代码

#include <cmath>
#include <cstdio>

typedef long long int64;

#ifdef ONLINE_JUDGE
const int MAXB = 8000000;
#else
const int MAXB = 10000 + 10;
#endif

const int MOD = 20190601;

int64 n;
int B;

int64 d[MAXB];
int dn;
int p[MAXB], psum[MAXB], pn;

inline int qmod(int x) { return (x >= MOD) ? (x - MOD) : x; }
void linear_sleve() {
    static bool notp[MAXB];
    int i, j;

    notp[1] = true;
    for (i = 2; i <= B; i++) {
        if (!notp[i]) {
            pn++;
            p[pn] = i;
            psum[pn] = qmod(psum[pn - 1] + i);
        }
        for (j = 1; i * p[j] <= B; j++) {
            notp[i * p[j]] = true;
            if (!(i % p[j]))
                break;
        }
    }
}
int zz_sleve() {
    static int f1[MAXB], f2[MAXB], last[MAXB];
    int i, j, pos;
    int64 x;
    int ans = 0;

    for (x = n; x; x = n / (d[dn] + 1)) {
        dn++;
        d[dn] = n / x;
    }
    for (i = 1; i <= dn; i++) {
        f1[i] = d[i] % MOD;
        f2[i] = f1[i] * (f1[i] + 1ll) / 2 % MOD;
    }
    for (i = 1; i <= pn; i++) {
        for (j = dn; j; j--) {
            x = d[j] / p[i];
            if (x < p[i])
                break;

            pos = (x <= B) ? x : (dn - n / x + 1);
            f1[pos] = qmod(MOD + f1[pos] - (i - 1 - last[pos]));
            f2[pos] = qmod(f2[pos] + qmod(MOD + psum[last[pos]] - psum[i - 1]));
            last[pos] = i - 1;

            f1[j] = qmod(MOD + f1[j] - f1[pos]);
            f2[j] = qmod(MOD + f2[j] - 1ll * f2[pos] * p[i] % MOD);
            last[j] = i;
        }

        x = n / p[i];
        while (x) {
            pos = (x <= B) ? x : (dn - n / x + 1);

            if (d[pos] < p[i])
                f1[pos] = 1;
            else {
                f1[pos] = qmod(MOD + f1[pos] - (i - last[pos]));
                f2[pos] = qmod(f2[pos] + qmod(MOD + psum[last[pos]] - psum[i]));
                last[pos] = i;
            }

            ans = (ans + 1ll * p[i] * f1[pos]) % MOD;
            x /= p[i];
        }
    }
    ans = qmod(MOD + ans - n % MOD);
    ans = qmod(ans + qmod(MOD + f2[dn] - f1[dn]));
    ans = qmod(ans + qmod(MOD - pn + 1));
    return ans;
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("6625.in", "r", stdin);
    freopen("6625.out", "w", stdout);
#endif

    scanf("%lld", &n);
    B = sqrt(n);
    linear_sleve();
    printf("%d\n", zz_sleve());
    return 0;
}
